89 const char* searchTarget = "ssdp:all", uint32_t minWaitMs = 3000,
90 uint32_t maxWaitMs = 60000) {
91 DlnaLogger.log(DlnaLogLevel::Info, "DLNADevice::begin");
92 search_target = searchTarget;
93 is_active = true;
94 p_udp = &udp;
95 p_http = &http;
96
97 if (p_http_server && http_server_port > 0 && eventCallback != nullptr) {
98 // handle server requests
99 if (!p_http_server->begin(http_server_port)) {
100 DlnaLogger.log(DlnaLogLevel::Error, "HttpServer begin failed");
101 return false;
102 }
103 }
104
105 // setup multicast UDP
106 if (!(p_udp->begin(DLNABroadcastAddress))) {
107 DlnaLogger.log(DlnaLogLevel::Error, "UDP begin failed");
108 return false;
109 }
110
111 // Send MSearch request via UDP. Use maxWaitMs as the emission window.
112 MSearchSchedule* search = new MSearchSchedule(DLNABroadcastAddress, searchTarget);
113
114 // ensure min <= max
115 if (minWaitMs > maxWaitMs) minWaitMs = maxWaitMs;
116 search->end_time = millis() + maxWaitMs;
117 search->repeat_ms = msearch_repeat_ms;
118 search->active = true;
119 scheduler.add(search);
120
121 // if maxWaitMs > 0 we will block here and process events. We guarantee
122 // we wait at least minWaitMs before returning; we stop waiting after
123 // maxWaitMs or earlier if stopWhenFound is true AND minWaitMs has elapsed
124 uint64_t start = millis();
125 uint64_t minEnd = start + minWaitMs;
126 uint64_t maxEnd = start + maxWaitMs;
127 while (millis() < maxEnd) {
128 // if a device is found and we've satisfied the minimum wait, we can
129 // return early (we always allow early return after minWaitMs when a
130 // device has been discovered)
131 if (devices.size() > 0 && millis() >= minEnd) break;
132 loop();
133 }
134 // If we exited early because a device was found, deactivate the MSearch
135 // schedule so it will stop repeating. The scheduler will clean up
136 // inactive schedules on its next pass.
137 if (devices.size() > 0 && search != nullptr) {
138 search->active = false;
139 }
140
141 DlnaLogger.log(DlnaLogLevel::Info,
142 "Control Point started with %d devices found",
143 devices.size());
144 return devices.size() > 0;
145 }

250 bool loop() {
251 if (!is_active) return false;
252 DLNAControlPointRequestParser parser;
253
254 // process UDP requests
255 RequestData req = p_udp->receive();
256 if (req && scheduler.isMSearchActive()) {
257 Schedule* schedule = parser.parse(req);
258 if (schedule != nullptr) {
259 // handle NotifyReplyCP
260 if (StrView(schedule->name()).equals("NotifyReplyCP")) {
261 NotifyReplyCP& notify_schedule = *(NotifyReplyCP*)schedule;
262 // notify_schedule.callback = processDevice;
263 processDevice(notify_schedule);
264 }
265 // handle MSearchReplyCP (HTTP/1.1 200 OK responses to M-SEARCH)
266 if (StrView(schedule->name()).equals("MSearchReplyCP")) {
267 MSearchReplyCP& ms_schedule = *(MSearchReplyCP*)schedule;
268 // Process M-SEARCH replies immediately: they contain LOCATION URLs
269 // pointing to device descriptions. Add the device and free the
270 // temporary schedule object.
271 processMSearchReply(ms_schedule);
272 }
273 }
274 }
275
276 // execute scheduled udp replys
277 scheduler.execute(*p_udp);
278
279 if (p_http_server) {
280 // handle server requests
281 bool rc = p_http_server->doLoop();
282 }
283
284 // be nice, if we have other tasks
285 delay(5);
286 return true;
287 }

433 void attachHttpServer(HttpServer& server) {
434 DlnaLogger.log(DlnaLogLevel::Debug,
435 "DLNAControlPointMgr::attachHttpServer");
436 p_http_server = &server;
437 // register handler at the local path. If local_url is not set we use
438 // a default path "/dlna/events"
439 const char* path =
440 StrView(local_url.url()).isEmpty() ? "/dlna/events" : local_url.path();

442 // handler lambda: reads SID and body, forwards to parseAndDispatchEvent
443 auto notifyHandler = [](HttpServer* server, const char* requestPath,
444 HttpRequestHandlerLine* hl) {
445 // The DLNAControlPointMgr instance is passed as context[0]
446 DLNAControlPointMgr* cp = nullptr;
447 if (hl->contextCount > 0)
448 cp = static_cast<DLNAControlPointMgr*>(hl->context[0]);
449 if (!cp) {
450 server->replyNotFound();
451 return;
452 }
453
454 // read headers
455 HttpRequestHeader& req = server->requestHeader();
456 const char* sid = req.get("SID");

464 tmp.xml = body.c_str();

466 cp->parseAndDispatchEvent(tmp);

471 void* ctx[1];
472 ctx[0] = this;
473 server.on(path, T_POST, notifyHandler, ctx, 1);

477 static bool processDevice(NotifyReplyCP& data) {
478 DlnaLogger.log(DlnaLogLevel::Debug, "DLNAControlPointMgr::processDevice");
479
480 Str& nts = data.nts;
481 if (nts.equals("ssdp:byebye")) {
482 selfDLNAControlPoint->processBye(nts);
483 return true;
484 }
485 if (nts.equals("ssdp:alive")) {
486 bool select = selfDLNAControlPoint->matches(data.usn.c_str());
487 DlnaLogger.log(DlnaLogLevel::Debug, "addDevice: %s -> %s",
488 data.usn.c_str(), select ? "added" : "filtered");
489 if (!select) return false;
490 // If we already know the device UDN, avoid fetching the device xml
491 // repeatedly from multiple Location URLs. Extract the UDN part from
492 // the USN (up to the first "::" if present) and check the existing
493 // device list.
494 const char* usn_c = data.usn.c_str();
495 const char* sep = strstr(usn_c, "::");
496 int udn_len = sep ? (int)(sep - usn_c) : (int)strlen(usn_c);
497 bool known = false;
498 for (auto &dev : selfDLNAControlPoint->devices) {
499 const char* known_udn = dev.getUDN();
500 if (known_udn && strncmp(known_udn, usn_c, udn_len) == 0 &&
501 (int)strlen(known_udn) == udn_len) {
502 DlnaLogger.log(DlnaLogLevel::Debug,
503 "Device '%s' already known (skip GET)", known_udn);
504 dev.setActive(true);
505 known = true;
506 break;
507 }
508 }
509 if (!known) {
510 Url url{data.location.c_str()};
511 selfDLNAControlPoint->addDevice(url);
512 }
513 return true;
514 }
515 return false;
516 }

519 static bool processMSearchReply(MSearchReplyCP& data) {
520 DlnaLogger.log(DlnaLogLevel::Debug,
521 "DLNAControlPointMgr::processMSearchReply");
522 // data.location contains the device description URL
523 if (data.location.isEmpty()) return false;
524 DlnaLogger.log(DlnaLogLevel::Info, "MSearchReply -> add device: %s",
525 data.location.c_str());
526 // If the reply includes a USN we can avoid fetching multiple
527 // different LOCATION URLs for the same device (different network
528 // interfaces) by checking for the UDN part in known devices.
529 const char* usn_c = data.usn.c_str();
530 if (usn_c && *usn_c) {
531 const char* sep = strstr(usn_c, "::");
532 int udn_len = sep ? (int)(sep - usn_c) : (int)strlen(usn_c);
533 for (auto &dev : selfDLNAControlPoint->devices) {
534 const char* known_udn = dev.getUDN();
535 if (known_udn && strncmp(known_udn, usn_c, udn_len) == 0 &&
536 (int)strlen(known_udn) == udn_len) {
537 DlnaLogger.log(DlnaLogLevel::Debug,
538 "MSearchReply: device '%s' already known (skip GET)",
539 known_udn);
540 dev.setActive(true);
541 return true;
542 }
543 }
544 }
545
546 Url url{data.location.c_str()};
547 selfDLNAControlPoint->addDevice(url);
548 return true;
549 }

776 r->out->arguments.push_back(arg);
777

780 // SID is not applicable here (action response), pass nullptr
781 r->self->eventCallback(r->self->reference, "", namePtr, valuePtr);
782 }
783 };
784

786 parser.setReference(&cbref);
787 parser.parse();
788 }
789 }
790 }
791
792 return result;
793 }
