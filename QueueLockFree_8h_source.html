32 void resize(size_t capacity) {
33 // round up to next power-of-two and compute mask/value
34 capacity_mask = capacity - 1;
35 for (size_t i = 1; i <= sizeof(void*) * 4; i <<= 1)
36 capacity_mask |= capacity_mask >> i;
