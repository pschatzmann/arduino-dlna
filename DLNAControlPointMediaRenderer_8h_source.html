501 // Helper: parse DLNA time string (HH:MM:SS, MM:SS) to milliseconds
502 static unsigned long parseTimeToMs(const char* t) {
503 if (t == nullptr) return 0;
504 int h = 0, m = 0, s = 0;
505 int parts = 0;
506 // Count colons
507 const char* p = t;
508 int colonCount = 0;
509 while (*p) {
510 if (*p == ':') colonCount++;
511 p++;
512 }
513 if (colonCount == 2) {
514 parts = sscanf(t, "%d:%d:%d", &h, &m, &s);
515 } else if (colonCount == 1) {
516 parts = sscanf(t, "%d:%d", &m, &s);
517 h = 0;
518 } else {
519 // maybe it's seconds only
520 parts = sscanf(t, "%d", &s);
521 }
522 if (parts <= 0) return 0;
523 unsigned long total =
524 (unsigned long)h * 3600UL + (unsigned long)m * 60UL + (unsigned long)s;
525 return total * 1000UL;
526 }
