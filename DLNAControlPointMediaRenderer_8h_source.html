478 bool getProtocalInfo(
479 std::function<void(const char* entry, ProtocolRole role)> cb) {
480 // Custom streaming implementation because the returned CSV can be very
481 // large and must not be buffered fully in memory.
482 if (!cb) return false;
483
484 DLNAServiceInfo& svc =
485 p_mgr->getService("urn:upnp-org:serviceId:ConnectionManager");
486 if (!svc) {
487 DlnaLogger.log(DlnaLogLevel::Error, "No ConnectionManager service found");
488 return false;
489 }
490
491 // Build POST URL using control point helper which normalizes slashes
492 DLNADeviceInfo& device = p_mgr->getDevice();
493 char url_buffer[256] = {0};
494 const char* suffix = nullStr(svc.control_url);
495 Url post_url{
496 p_mgr->getUrl(device, suffix, url_buffer, (int)sizeof(url_buffer))};
497
498 // Send request using chunked upload to avoid buffering the SOAP body
499 p_http->stop();
500 char soapaction[200];
501 snprintf(soapaction, sizeof(soapaction), "\"%s#GetProtocolInfo\"",
502 svc.service_type);
503 p_http->request().put("SOAPACTION", soapaction);
504
505 int rc = p_http->postChunked(
506 post_url, DLNAControlPointMediaRenderer::writeGetProtocolInfoBody,
507 "text/xml");
508 if (rc != 200) {
509 DlnaLogger.log(DlnaLogLevel::Error,
510 "GetProtocolInfo POST failed with rc %d", rc);
511 return false;
512 }
513
514 // Stream-parse response using centralized parser (low-memory)
515 Client* cli = p_http->client();
516 if (!cli) return false;
517 return XMLProtocolInfoParser::parseFromClient(cli, cb);
518 }

587 // Helper: parse DLNA time string (HH:MM:SS, MM:SS) to milliseconds
588 static unsigned long parseTimeToMs(const char* t) {
589 if (t == nullptr) return 0;
590 int h = 0, m = 0, s = 0;
591 int parts = 0;
592 // Count colons
593 const char* p = t;
594 int colonCount = 0;
595 while (*p) {
596 if (*p == ':') colonCount++;
597 p++;
598 }
599 if (colonCount == 2) {
600 parts = sscanf(t, "%d:%d:%d", &h, &m, &s);
601 } else if (colonCount == 1) {
602 parts = sscanf(t, "%d:%d", &m, &s);
603 h = 0;
604 } else {
605 // maybe it's seconds only
606 parts = sscanf(t, "%d", &s);
607 }
608 if (parts <= 0) return 0;
609 unsigned long total =
610 (unsigned long)h * 3600UL + (unsigned long)m * 60UL + (unsigned long)s;
611 return total * 1000UL;
612 }
