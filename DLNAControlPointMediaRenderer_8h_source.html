467 bool getProtocalInfo(

469 if (!cb) return false;
470
471 DLNAServiceInfo& svc =
472 p_mgr->getService("urn:upnp-org:serviceId:ConnectionManager");
473 if (!svc) {
474 DlnaLogger.log(DlnaLogLevel::Error, "No ConnectionManager service found");
475 return false;

477
478 // build action and enqueue
479 ActionRequest act(svc, "GetProtocolInfo");
480 p_mgr->addAction(act);
481 bool ok = false;
482
483 // processor will be invoked with the live Client and the ActionReply so
484 // callers can handle the response stream themselves. We simply stream-
485 // parse and forward each parsed entry to the provided callback.
486 XMLCallback processor =
487 [&cb, &ok](Client& client, ActionReply& reply) {
488 ok = XMLProtocolInfoParser::parse(client, cb);
489 reply.clear();
490 reply.setValid(true);
491 };
492
493 // execute the queued action and use our processor to handle the reply
494 p_mgr->executeActions(processor);
495 return ok;

547 // Helper: parse DLNA time string (HH:MM:SS, MM:SS) to milliseconds
548 static unsigned long parseTimeToMs(const char* t) {
549 if (t == nullptr) return 0;
550 int h = 0, m = 0, s = 0;
551 int parts = 0;
552 // Count colons
553 const char* p = t;
554 int colonCount = 0;
555 while (*p) {
556 if (*p == ':') colonCount++;
557 p++;
558 }
559 if (colonCount == 2) {
560 parts = sscanf(t, "%d:%d:%d", &h, &m, &s);
561 } else if (colonCount == 1) {
562 parts = sscanf(t, "%d:%d", &m, &s);
563 h = 0;
564 } else {
565 // maybe it's seconds only
566 parts = sscanf(t, "%d", &s);
567 }
568 if (parts <= 0) return 0;
569 unsigned long total =
570 (unsigned long)h * 3600UL + (unsigned long)m * 60UL + (unsigned long)s;
571 return total * 1000UL;
572 }
