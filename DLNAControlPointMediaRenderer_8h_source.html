527 bool getProtocalInfo(
528 std::function<void(const char* entry, ProtocolRole role)> cb) {
529 if (!cb) return false;
530
531 DLNAServiceInfo& svc =
532 p_mgr->getService("urn:upnp-org:serviceId:ConnectionManager");
533 if (!svc) {
534 DlnaLogger.log(DlnaLogLevel::Error, "No ConnectionManager service found");
535 return false;
536 }
537
538 // build action and enqueue
539 ActionRequest act(svc, "GetProtocolInfo");
540 p_mgr->addAction(act);
541 bool ok = false;
542
543 // processor will be invoked with the live Client and the ActionReply so
544 // callers can handle the response stream themselves. We simply stream-
545 // parse and forward each parsed entry to the provided callback.
546 XMLCallback processor = [&cb, &ok](Client& client, ActionReply& reply) {
547 ok = XMLProtocolInfoParser::parse(client, cb);
548 reply.clear();
549 reply.setValid(true);
550 };
551
552 // execute the queued action and use our processor to handle the reply
553 p_mgr->executeActions(processor);
554 return ok;

608 // Helper: parse DLNA time string (HH:MM:SS, MM:SS) to milliseconds
609 static unsigned long parseTimeToMs(const char* t) {
610 if (t == nullptr) return 0;
611 int h = 0, m = 0, s = 0;
612 int parts = 0;
613 // Count colons
614 const char* p = t;
615 int colonCount = 0;
616 while (*p) {
617 if (*p == ':') colonCount++;
618 p++;
619 }
620 if (colonCount == 2) {
621 parts = sscanf(t, "%d:%d:%d", &h, &m, &s);
622 } else if (colonCount == 1) {
623 parts = sscanf(t, "%d:%d", &m, &s);
624 h = 0;
625 } else {
626 // maybe it's seconds only
627 parts = sscanf(t, "%d", &s);
628 }
629 if (parts <= 0) return 0;
630 unsigned long total =
631 (unsigned long)h * 3600UL + (unsigned long)m * 60UL + (unsigned long)s;
632 return total * 1000UL;
633 }
