117 int findGt(const char* s, int start, int len) {
118 int gt = -1;
119 bool inQuote = false;
120 char qchar = 0;
121 for (int i = start + 1; i < len; ++i) {
122 char c = s[i];

124 inQuote = true;
125 qchar = c;
126 } else if (inQuote && c == qchar) {
127 inQuote = false;
128 } else if (!inQuote && c == '>') {
129 gt = i;

158 void handleStartTag(const char* s, int lt, int gt) {
159 int nameStart = lt + 1;
160 while (nameStart < gt && isspace((unsigned char)s[nameStart])) nameStart++;
161 int nameEnd = nameStart;
162 while (nameEnd < gt && !isspace((unsigned char)s[nameEnd]) &&
163 s[nameEnd] != '/' && s[nameEnd] != '>')
164 nameEnd++;
165 if (nameEnd > nameStart) {
166 node_name.copyFrom(s + nameStart, nameEnd - nameStart);
167 path.push_back(node_name);
168 int contentStart = gt + 1;
169 contentStarts.push_back(contentStart);
170 // extract raw attribute text (between name end and tag end), exclude
171 // trailing '/' for self-closing tags
172 int attrStart = nameEnd;
173 int attrEnd = gt;
174 // back up one if tag ends with '/>'
175 int back = gt - 1;
176 while (back > lt && isspace((unsigned char)s[back])) back--;
177 if (back > lt && s[back] == '/') attrEnd = back;
178 // trim leading/trailing whitespace for attributes
179 while (attrStart < attrEnd && isspace((unsigned char)s[attrStart]))
180 attrStart++;
181 while (attrEnd > attrStart && isspace((unsigned char)s[attrEnd - 1]))
182 attrEnd--;
183 if (attrEnd > attrStart)
184 last_attributes.copyFrom(s + attrStart, attrEnd - attrStart);

208 void invokeCallback(Str& nodeName, Vector<Str>& fullPath, Str& text,
209 Str& attributes, int start, int len) {
210 if (!callback) return;
211 if (report_text_only && text.isEmpty()) return;
212 Vector<Str> ancestorPath;
213 int ancCount = fullPath.size() > 0 ? (int)fullPath.size() - 1 : 0;
214 for (int i = 0; i < ancCount; ++i) {
215 // Make a fresh copy of the ancestor name to avoid any aliasing
216 // or lifetime issues with parser-owned storage.
217 ancestorPath.push_back(Str(fullPath[i].c_str()));
218 }
219 callback(nodeName, ancestorPath, text, attributes, start, len, reference);
220 }
221 // Run the parser. This is a small, forgiving parser suitable for the
222 // embedded use-cases in this project (DIDL fragments, simple SCPD parsing).
223 // It is not a full XML validator but handles start/end tags, comments,
224 // processing instructions and self-closing tags. For each text node that
225 // contains non-whitespace characters the provided callback is invoked with
226 // the text and the current element path.
227 void do_parse() {
228 const char* s = str_view.c_str();
229 int len = str_view.length();
230 int pos = 0;
231
232 while (pos < len) {
233 int lt = str_view.indexOf('<', pos);
234 if (lt < 0) break;
235
236 // Handle text between pos and lt
237 if (lt > pos) {
238 int ts = pos;
239 int te = lt;
240 while (ts < te && isspace((unsigned char)s[ts])) ts++;
241 while (te > ts && isspace((unsigned char)s[te - 1])) te--;
242 emitTextSegment(s, ts, te);
243 }
244
245 // comment or processing instruction handling
246 int newPos = handleCommentOrPI(lt, s, len);
247 if (newPos != lt) {
248 pos = newPos;
249 continue;
250 }
251
252 // find closing '>' (respect quotes)
253 int gt = findGt(s, lt, len);
254 if (gt < 0) break; // malformed
255
256 // end tag
257 if (lt + 1 < len && s[lt + 1] == '/') {
258 pos = handleEndTag(s, lt, gt);
259 continue;
260 }
261
262 // start tag (or self-closing)
263 handleStartTag(s, lt, gt);
264
265 // callback for the tag itself
266 emitTagSegment(s, lt, gt);
267
268 // detect self-closing and pop if needed
269 bool selfClosing = false;
270 int back = gt - 1;
271 while (back > lt && isspace((unsigned char)s[back])) back--;
272 if (back > lt && s[back] == '/') selfClosing = true;
273
274 pos = gt + 1;
275
276 if (selfClosing && path.size() > 0) {
277 path.erase(path.size() - 1);
278 if (contentStarts.size() > 0)
279 contentStarts.erase(contentStarts.size() - 1);
280 }
281 }
