160 int findGt(const char* s, int start, int len) {
161 int gt = -1;
162 bool inQuote = false;
163 char qchar = 0;
164 for (int i = start + 1; i < len; ++i) {
165 char c = s[i];
166 if (!inQuote && (c == '"' || c == '\'')) {
167 inQuote = true;
168 qchar = c;
169 } else if (inQuote && c == qchar) {
170 inQuote = false;
171 } else if (!inQuote && c == '>') {
172 gt = i;
173 break;
174 }
175 }
176 return gt;
177 }

204 void handleStartTag(const char* s, int lt, int gt) {
205 // Clear previous attributes first to avoid carrying attributes from a
206 // previously parsed tag into the current element when the current
207 // tag contains no attributes. Use clear() so the underlying char buffer
208 // is zero-terminated and not accidentally read as a C-string by callers
209 // that use attributes.c_str().
210 last_attributes.clear();
211 int nameStart = lt + 1;
212 while (nameStart < gt && isspace((unsigned char)s[nameStart])) nameStart++;
213 int nameEnd = nameStart;
214 while (nameEnd < gt && !isspace((unsigned char)s[nameEnd]) &&
215 s[nameEnd] != '/' && s[nameEnd] != '>')
216 nameEnd++;
217
218 if (nameEnd > nameStart) {
219 node_name.copyFrom(s + nameStart, nameEnd - nameStart);
220 path.push_back(node_name);
221 int contentStart = gt + 1;
222 contentStarts.push_back(contentStart);
223 // extract raw attribute text (between name end and tag end), exclude
224 // trailing '/' for self-closing tags
225 int attrStart = nameEnd;
226 int attrEnd = gt;
227 // back up one if tag ends with '/>'
228 int back = gt - 1;
229 while (back > lt && isspace((unsigned char)s[back])) back--;
230 if (back > lt && s[back] == '/') attrEnd = back;
231 // trim leading/trailing whitespace for attributes
232 while (attrStart < attrEnd && isspace((unsigned char)s[attrStart]))
233 attrStart++;
234 while (attrEnd > attrStart && isspace((unsigned char)s[attrEnd - 1]))
235 attrEnd--;
236 if (attrEnd > attrStart) {
237 last_attributes.copyFrom(s + attrStart, attrEnd - attrStart);
238 } else {
239 last_attributes.clear();
240 }
241 }

244 int handleCommentOrPI(int lt, const char* s, int len) {
245 // comment <!-- ... -->

259 // Helper: invoke the registered callback but pass a path that excludes the
260 // current node (only ancestor elements). This keeps the `nodeName`
261 // parameter as the current element while `path` contains only parents.
262 bool invokeCallback(Str& nodeName, Vector<Str>& fullPath, Str& text,
263 Str& attributes, int start, int len) {
264 if (!callback) return false;
265 if (report_text_only && text.isEmpty()) return false;
266 Vector<Str> ancestorPath;
267 int ancCount = fullPath.size() > 0 ? (int)fullPath.size() - 1 : 0;
268 for (int i = 0; i < ancCount; ++i) {
269 // Make a fresh copy of the ancestor name to avoid any aliasing
270 // or lifetime issues with parser-owned storage.
271 ancestorPath.push_back(Str(fullPath[i].c_str()));
272 }
273 callback(nodeName, ancestorPath, text, attributes, start, len, reference);
274 return true;
275 }

277 // Run the parser. This is a small, forgiving parser suitable for the
278 // embedded use-cases in this project (DIDL fragments, simple SCPD parsing).
279 // It is not a full XML validator but handles start/end tags, comments,
280 // processing instructions and self-closing tags. For each text node that
281 // contains non-whitespace characters the provided callback is invoked with
282 // the text and the current element path.
283 void do_parse() {
284 const char* s = str_view.c_str();
285 int len = str_view.length();
286 int pos = 0;
287
288 while (pos < len) {
289 int lt = str_view.indexOf('<', pos);
290 if (lt < 0) break;
291
292 // Handle text between pos and lt
293 if (lt > pos) {
294 int ts = pos;
295 int te = lt;
296 while (ts < te && isspace((unsigned char)s[ts])) ts++;
297 while (te > ts && isspace((unsigned char)s[te - 1])) te--;
298 emitTextSegment(s, ts, te);
299 }
300
301 // comment or processing instruction handling
302 int newPos = handleCommentOrPI(lt, s, len);
303 if (newPos != lt) {
304 pos = newPos;
305 continue;
306 }
307
308 // find closing '>' (respect quotes)
309 int gt = findGt(s, lt, len);
310 if (gt < 0) break; // malformed
311
312 // end tag
313 if (lt + 1 < len && s[lt + 1] == '/') {
314 pos = handleEndTag(s, lt, gt);
315 continue;
316 }
317
318 // start tag (or self-closing)
319 handleStartTag(s, lt, gt);
320
321 // callback for the tag itself
322 emitTagSegment(s, lt, gt);
323
324 // detect self-closing and pop if needed
325 bool selfClosing = false;
326 int back = gt - 1;
327 while (back > lt && isspace((unsigned char)s[back])) back--;
328 if (back > lt && s[back] == '/') selfClosing = true;
329
330 pos = gt + 1;

332 if (selfClosing && path.size() > 0) {
333 path.erase(path.size() - 1);
334 if (contentStarts.size() > 0)

336 }
337 }

340 // Incremental single-callback parser. It will resume from `parsePos`
341 // and advance `parsePos` past the fragment it invoked (or to the end
342 // if nothing more is parseable). Returns true if a callback was
343 // invoked.
344 bool do_parse_single() {
345 const char* s = str_view.c_str();
346 int len = str_view.length();

348 bool result = false;
349

352 if (lt < 0) break;
353
354 // Handle text between pos and lt
355 if (lt > pos) {
356 int ts = pos;
357 int te = lt;

359 while (te > ts && isspace((unsigned char)s[te - 1])) te--;
360 if (te > ts && callback) {

365 // the parser text as-is.

375 if (newPos != lt) {
