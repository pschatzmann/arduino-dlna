352 size_t printService(Print& out, void* srv) {
353 XMLPrinter xp(out);
354 size_t result = 0;
355 char buffer[DLNA_MAX_URL_LEN] = {0};
356 StrView url(buffer, DLNA_MAX_URL_LEN);
357 DLNAServiceInfo* service = (DLNAServiceInfo*)srv;
358 result += xp.printNode("serviceType", service->service_type);
359 result += xp.printNode("serviceId", service->service_id);
360 result +=
361 xp.printNode("SCPDURL", url.buildPath(base_url, service->scpd_url));
362 result += xp.printNode("controlURL",
363 url.buildPath(base_url, service->control_url));
364 if (is_subcription_active)
365 result += xp.printNode("eventSubURL",
366 url.buildPath(base_url, service->event_sub_url));
367 else
368 result += xp.printf("<eventSubURL/>");
369
370 return result;
371 }
