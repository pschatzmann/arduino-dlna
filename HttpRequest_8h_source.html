193 virtual int process(TinyMethodID action, Url& url, const char* mime,
194 const char* data, int len = -1) {
195 if (!connected()) {
196 DlnaLogger.log(DlnaLogLevel::Info, "Connecting to host %s port %d",
197 url.host(), url.port());
198
199 connect(url.host(), url.port());
200 }
201
202 if (!connected()) {
203 DlnaLogger.log(DlnaLogLevel::Info, "Connected: %s",
204 connected() ? "true" : "false");
205 return -1;
206 }
207
208 if (host_name.isEmpty()) {
209 host_name = url.host();
210 host_name.add(":");
211 host_name.add(url.port());
212 }
213
214 request_header.setValues(action, url.path());
215 if (len == -1 && data != nullptr) {
216 len = strlen(data);
217 }
218 if (len > 0) {
219 request_header.put(CONTENT_LENGTH, len);
220 }
221 if (!host_name.isEmpty()) {
222 request_header.put(HOST_C, host_name.c_str());
223 }
224 if (agent != nullptr) {
225 request_header.put(USER_AGENT, agent);
226 }
227 if (accept_encoding != nullptr) {
228 request_header.put(ACCEPT_ENCODING, accept_encoding);
229 }
230 if (mime != nullptr) {
231 request_header.put(CONTENT_TYPE, mime);
232 }
233
234 request_header.put(CONNECTION, connection);
235 request_header.put(ACCEPT, accept);
236
237 request_header.write(*client_ptr);
238
239 if (len > 0) {
240 DlnaLogger.log(DlnaLogLevel::Info, "process - writing data: %d bytes",
241 len);
242 client_ptr->write((const uint8_t*)data, len);
243 }
244
245 reply_header.read(*client_ptr);
246
247 // if we use chunked tranfer we need to read the first chunked length
248 if (reply_header.isChunked()) {

257 virtual int process(TinyMethodID method, Url& url, size_t len,
258 std::function<size_t(Print&, void*)> writer,
259 const char* mime = nullptr, void* ref = nullptr) {
260 DlnaLogger.log(DlnaLogLevel::Info, "%s %s", methods[method], url.url());
261
262 if (!connected()) {
263 DlnaLogger.log(DlnaLogLevel::Info,
264 "HttpRequest::connecting to host %s port %d", url.host(),
265 url.port());
266 connect(url.host(), url.port());
267 }
268
269 if (!connected()) {
270 DlnaLogger.log(DlnaLogLevel::Info, "Connected: %s",
271 connected() ? "true" : "false");
272 return -1;
273 }
274
275 if (host_name.isEmpty()) {
276 host_name = url.host();
277 host_name.add(":");
278 host_name.add(url.port());
279 }
280
281 // prepare request header
282 request_header.setValues(method, url.path());
283 if (!host_name.isEmpty()) {
284 request_header.put(HOST_C, host_name.c_str());
285 }
286 if (agent != nullptr) request_header.put(USER_AGENT, agent);
287 if (accept_encoding != nullptr)
288 request_header.put(ACCEPT_ENCODING, accept_encoding);
289 if (mime != nullptr) request_header.put(CONTENT_TYPE, mime);
290 if (len > 0) request_header.put(CONTENT_LENGTH, len);
291
292 request_header.put(CONNECTION, connection);
293 request_header.put(ACCEPT, accept);
294
295 // write request header to client
296 request_header.write(*client_ptr);
297
298 // write callback (writer returns number of bytes written)
299 if (writer) {
300 size_t written = writer(*client_ptr, ref);
301#if DLNA_LOG_XML
302 writer(Serial, ref);
303#endif
304 if (written != len) {
305 DlnaLogger.log(DlnaLogLevel::Error,
306 "HttpRequest wrote %d bytes: expected %d", written, len);
307 }
308 }
309
310 // read reply header and prepare chunk reader if needed
311 reply_header.read(*client_ptr);
312 if (reply_header.isChunked()) {
313 chunk_reader.open(*client_ptr);
314 }
315
316 return reply_header.statusCode();
317 }
