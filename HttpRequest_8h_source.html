77 if (!connected()) {

82 if (host_name.isEmpty()) {

84 host_name.add(":");

87
88 // prepare request header
89 request_header.setValues(T_POST, url.path());
90 if (!host_name.isEmpty()) {
91 request_header.put(HOST_C, host_name.c_str());
92 }
93 if (agent != nullptr) request_header.put(USER_AGENT, agent);
94 if (accept_encoding != nullptr) request_header.put(ACCEPT_ENCODING, accept_encoding);
95 if (mime != nullptr) request_header.put(CONTENT_TYPE, mime);
96 request_header.put(CONNECTION, connection);
97 request_header.put(ACCEPT, accept);
98
99 // write request header to client
100 request_header.write(*client_ptr);
101
102 // stream body via ChunkPrint which emits proper chunk frames
103 if (writer) writer(*client_ptr);
104
105 // read reply header and prepare chunk reader if needed
106 reply_header.read(*client_ptr);
107 if (reply_header.isChunked()) {
108 chunk_reader.open(*client_ptr);
109 }
110
111 return reply_header.statusCode();

261
