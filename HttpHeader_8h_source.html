235 // reads the full header from the request (stream)
236 void read(Client& in) {
237 DlnaLogger.log(DlnaLogLevel::Debug, "HttpHeader::read");
238 // remove all existing value
239 clear();
240
241 char line[MaxHeaderLineLength];
242 if (in.connected()) {
243 delay(200);
244 if (in.available() == 0) {
245 DlnaLogger.log(DlnaLogLevel::Warning, "Waiting for data...");
246 uint64_t end = millis() + in.getTimeout();
247 while (in.available() == 0) {
248 delay(100);
249 if (millis()>end) break;
250 }
251 }
252 readLine(in, line, MaxHeaderLineLength);
253 parse1stLine(line);
254 while (in.available()) {
255 readLine(in, line, MaxHeaderLineLength);
256 StrView lineStr(line);
257 if (lineStr.isEmpty() || lineStr.isNewLine()) {
258 break;
259 }
260 if (isValidStatus() || isRedirectStatus()) {
261 lineStr.ltrim();
262 put(line);
263 }
264 }
265 }

455 void parse1stLine(const char* line) {
456 DlnaLogger.log(DlnaLogLevel::Debug, "HttpReplyHeader::parse1stLine %s", line);
457 StrView line_str(line);

459 int space2 = line_str.indexOf(' ', space1 + 1);
460

462 protocol_str.substrView(line_str, 0, space1);
463
464 // find response status code after the first space

467 status.substrView(line_str, space1 + 1, space2);
468 status_code = atoi(status_c);
469
