124 bool begin(DLNAHttpRequest& http, IUDPService& udp,
125 const char* searchTarget = "ssdp:all", uint32_t minWaitMs = 3000,
126 uint32_t maxWaitMs = 60000) {
127 DlnaLogger.log(DlnaLogLevel::Info, "DLNADevice::begin");
128 search_target = searchTarget;
129 is_active = true;
130 p_udp = &udp;
131 p_http = &http;
132
133 if (p_http_server && http_server_port > 0 && event_callback != nullptr) {
134 // handle server requests
135 if (!p_http_server->begin(http_server_port)) {
136 DlnaLogger.log(DlnaLogLevel::Error, "HttpServer begin failed");
137 return false;
138 }
139 }
140
141 // setup multicast UDP
142 if (!(p_udp->begin(DLNABroadcastAddress))) {
143 DlnaLogger.log(DlnaLogLevel::Error, "UDP begin failed");
144 return false;
145 }
146
147 // Send MSearch request via UDP. Use maxWaitMs as the emission window.
148 MSearchSchedule* search =
149 new MSearchSchedule(DLNABroadcastAddress, searchTarget);
150
151 // ensure min <= max
152 if (minWaitMs > maxWaitMs) minWaitMs = maxWaitMs;
153 search->end_time = millis() + maxWaitMs;
154 search->repeat_ms = msearch_repeat_ms;
155 search->active = true;
156 scheduler.add(search);
157
158 // if maxWaitMs > 0 we will block here and process events. We guarantee
159 // we wait at least minWaitMs before returning; we stop waiting after
160 // maxWaitMs or earlier if stopWhenFound is true AND minWaitMs has elapsed
161 uint64_t start = millis();
162 uint64_t minEnd = start + minWaitMs;
163 uint64_t maxEnd = start + maxWaitMs;
164 while (millis() < maxEnd) {
165 // if a device is found and we've satisfied the minimum wait, we can
166 // return early (we always allow early return after minWaitMs when a
167 // device has been discovered)
168 if (devices.size() > 0 && millis() >= minEnd) break;
169 loop();
170 }
171 // If we exited early because a device was found, deactivate the MSearch
172 // schedule so it will stop repeating. The scheduler will clean up
173 // inactive schedules on its next pass.
174 if (devices.size() > 0 && search != nullptr) {
175 search->active = false;
176 }
177
178 DlnaLogger.log(DlnaLogLevel::Info,
179 "Control Point started with %d devices found",
180 devices.size());
181
182 return devices.size() > 0;
183 }

235 bool loop() {
236 if (!is_active) return false;
237 DLNAControlPointRequestParser parser;
238
239 // process UDP requests
240 RequestData req = p_udp->receive();
241 if (req && scheduler.isMSearchActive()) {
242 Schedule* schedule = parser.parse(req);
243 if (schedule != nullptr) {
244 // handle NotifyReplyCP
245 if (StrView(schedule->name()).equals("NotifyReplyCP")) {
246 NotifyReplyCP& notify_schedule = *(NotifyReplyCP*)schedule;
247 // notify_schedule.callback = processDevice;
248 processDevice(notify_schedule);
249 }
250 // handle MSearchReplyCP (HTTP/1.1 200 OK responses to M-SEARCH)
251 if (StrView(schedule->name()).equals("MSearchReplyCP")) {
252 MSearchReplyCP& ms_schedule = *(MSearchReplyCP*)schedule;
253 // Process M-SEARCH replies immediately: they contain LOCATION URLs
254 // pointing to device descriptions. Add the device and free the
255 // temporary schedule object.
256 processMSearchReply(ms_schedule);
257 }
258 }
259 }
260
261 // execute scheduled udp replys
262 scheduler.execute(*p_udp);
263
264 if (p_http_server) {
265 // handle server requests
266 bool rc = p_http_server->doLoop();
267
268 // re-subscribe after timeout
269 if (event_callback != nullptr && subscribe_timeout > 0 &&
270 millis() >= subscribe_timeout) {
271 subscribeNotifications(getDevice(), subscribe_repeat_sec);
272 }
273 }
274
275 // be nice, if we have other tasks
276 delay(5);
277 return true;
278 }

467 static void notifyHandler(HttpServer* server, const char* requestPath,
468 HttpRequestHandlerLine* hl) {
469 // The DLNAControlPoint instance is passed as context[0]
470 DLNAControlPoint* cp = nullptr;
471 if (hl && hl->contextCount > 0) {
472 cp = static_cast<DLNAControlPoint*>(hl->context[0]);
473 }
474
475 if (!cp) {
476 DlnaLogger.log(DlnaLogLevel::Error, "No control point found");
477 server->replyNotFound();
478 return;
479 }
480
481 // read headers
482 HttpRequestHeader& req = server->requestHeader();
483 uint8_t buffer[200];
484 XMLParserPrint xml_parser;
485 Str node;
486 Str text;
487 Str attr;
488 Vector<Str> path;
489 bool active = false;
490 const char* sid = req.get("SID");
491
493 Client& client = server->client();
494 while (client.available()) {
495 int len = client.readBytes(buffer, sizeof(buffer));
496 xml_parser.write((const uint8_t*)buffer, len);
497 while (xml_parser.parse(node, path, text, attr)) {
498 // execute callback
499 if (active && !text.isEmpty()) {
500 DlnaLogger.log(DlnaLogLevel::Info, "Event: %s = %s", node.c_str(),
501 text.c_str());
502 // event callback
503 if (cp->event_callback != nullptr) {
504 cp->event_callback(cp->reference, sid, node.c_str(), text.c_str());
505 }
506 }
508 if (node.equals("e:property")) {
509 active = true;
510 }
511 }
512 }
513 // reply OK
514 server->replyOK();
515 }

576 static bool processMSearchReply(MSearchReplyCP& data) {
577 DlnaLogger.log(DlnaLogLevel::Debug,
578 "DLNAControlPointMgr::processMSearchReply");
579 // data.location contains the device description URL
580 if (data.location.isEmpty()) return false;
581 DlnaLogger.log(DlnaLogLevel::Info, "MSearchReply -> add device: %s",
582 data.location.c_str());
583 // If the reply includes a USN we can avoid fetching multiple
584 // different LOCATION URLs for the same device (different network
585 // interfaces) by checking for the UDN part in known devices.
586 const char* usn_c = data.usn.c_str();

588 const char* sep = strstr(usn_c, "::");
589 int udn_len = sep ? (int)(sep - usn_c) : (int)strlen(usn_c);
590 for (auto& dev : selfDLNAControlPoint->devices) {
591 const char* known_udn = dev.getUDN();
592 if (known_udn && strncmp(known_udn, usn_c, udn_len) == 0 &&
593 (int)strlen(known_udn) == udn_len) {
594 DlnaLogger.log(DlnaLogLevel::Debug,
595 "MSearchReply: device '%s' already known (skip GET)",
596 known_udn);
597 dev.setActive(true);
598 return true;
599 }
600 }
601 }
602
603 Url url{data.location.c_str()};
604 selfDLNAControlPoint->addDevice(url);
605 return true;

733 ActionReply& processActionHttpPost(Url& post_url, StrPrint& requestBody,
734 const char* soapAction) {
735 DlnaLogger.log(DlnaLogLevel::Debug, "DLNAControlPointMgr::processActionHttpPost");
736 XMLParserPrint xml_parser;
737 uint8_t buffer[200];
738 Str outNodeName;
739 Vector<Str> outPath;
740 Str outText;
741 Str outAttributes;
742 xml_parser.setExpandEncoded(true);
743
744 // set header and post
745 p_http->stop();
746 p_http->request().put("SOAPACTION", soapAction);
747 int rc = p_http->post(post_url, "text/xml", requestBody.c_str(),
748 requestBody.length());
749 if (rc != 200) {
750 p_http->stop();
751 reply.setValid(false);
752 return reply;
753 }
754
755 reply.setValid(true);
756 reply.arguments.clear();
757 // parse response
758 while (p_http->client()->available()) {
759 int len = p_http->client()->read(buffer, 200);
760 if (len > 0) {
761 // Serial.write(buffer, len);
762 xml_parser.write(buffer, len);
763 while (xml_parser.parse(outNodeName, outPath, outText, outAttributes)) {
764 // skip didl
765 if (!outNodeName.equals("Result")) {
766 Argument arg;
767 // persist the argument name in the control point's string registry
768 arg.name = strings.add((char*)outNodeName.c_str());
769 arg.value = outText;
770 if (!(outText.isEmpty() && outAttributes.isEmpty())) {
771 reply.addArgument(arg);
772 DlnaLogger.log(DlnaLogLevel::Info, "ActionReplay '%s': %s (%s)",
773 nullStr(outNodeName), nullStr(outText),
774 nullStr(outAttributes));
775 if (result_callback) {
776 result_callback(nullStr(arg.name, ""), nullStr(outText, ""),
777 nullStr(outAttributes, ""));
778 }
779 }
780 }
781 }
782 }
783 }
784 return reply;
785 }

815 DlnaLogger.log(DlnaLogLevel::Error, "Local URL not defined");
816 return false;
817 }
818 for (auto& service : device.getServices()) {
819 if (StrView(service.event_sub_url).isEmpty()) {
820 DlnaLogger.log(DlnaLogLevel::Warning,
821 "Service %s has no eventSubURL defined",
822 service.service_id);
823 continue;
824 }
825 if (!subscribeNotifications(service, timeoutSeconds)) {
826 DlnaLogger.log(DlnaLogLevel::Error, "Subscription to service %s failed",
827 service.service_id);
828 return false;
829 }
830 DlnaLogger.log(DlnaLogLevel::Info,
831 "Subscribed to service %s successfully",
832 service.service_id);
833 }
834 return true;
835 }
836
838 bool subscribeNotifications(DLNAServiceInfo& service,
839 int timoutSeconds = 1800) {
840 if (p_http_server == nullptr) {
841 DlnaLogger.log(
842 DlnaLogLevel::Error,
843 "HttpServer not defined - cannot subscribe to notifications");
844 return false;
845 }
846 DlnaLogger.log(DlnaLogLevel::Debug,
847 "DLNAControlPointMgr::subscribeNotifications");
848 DLNADeviceInfo& device = getDevice(service);
849
850 char url_buffer[200] = {0};
851 char seconds_txt[80] = {0};
852 Url url{getUrl(device, service.event_sub_url, url_buffer, 200)};
853 snprintf(seconds_txt, 80, "Second-%d", timoutSeconds);
854 p_http->request().put("NT", "upnp:event");
855 p_http->request().put("TIMEOUT", seconds_txt);
856 p_http->request().put("CALLBACK", local_url.url());
857 // post subscribe
858 int rc = p_http->subscribe(url);
859 DlnaLogger.log(DlnaLogLevel::Info, "Http rc: %s", rc);
860 if (rc == 200) {
861 subscribe_timeout = millis() + (timoutSeconds * 1000);
862 return true;
863 }
864
865 return false;
866 }
867};
