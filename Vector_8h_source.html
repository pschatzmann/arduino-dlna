299 void reset() {
300 // Ensure all elements are destroyed and the underlying buffer is freed.
301 // Previously this called deleteArray(p_data, size()) which used the
302 // current logical length (`len`) instead of the allocated buffer
303 // capacity (`max_capacity`). That caused the allocated buffer to
304 // remain unfreed when len was 0 (leading to a memory leak).
305 clear();
306 if (p_data != nullptr) {
307 // free the full allocated buffer
308 deleteArray(p_data, max_capacity);
309 p_data = nullptr;
310 max_capacity = 0;
311 len = 0;
312 }
313 }

321 void resize_internal(int newSize, bool copy, bool shrink = false) {
322 if (newSize <= 0) return;
323 if (newSize > max_capacity || this->p_data == nullptr || shrink) {
324 T *oldData = p_data;
325 int oldBufferLen = this->max_capacity;
326 p_data = newArray(newSize); // new T[newSize+1];
327 assert(p_data != nullptr);
328 this->max_capacity = newSize;
329 if (oldData != nullptr) {
330 if (copy && this->len > 0) {
331 // Element-wise copy using assignment/constructors. Using memmove
332 // here is unsafe for non-trivial types (it copies internal
333 // pointers/ownership and can lead to double-free). Do a proper
334 // element-wise copy so constructors/assignments run.
335 for (int j = 0; j < len; j++) {
336 p_data[j] = oldData[j];
337 }
338 }
339 if (shrink) {
340 cleanup(oldData, newSize, oldBufferLen);
341 }
342 deleteArray(oldData, oldBufferLen); // delete [] oldData;
343 }
344 }
345 }
