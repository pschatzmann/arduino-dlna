405 void reset() {
406 // Ensure all elements are destroyed and the underlying buffer is freed.
407 // Previously this called deleteArray(p_data, size()) which used the
408 // current logical length (`len`) instead of the allocated buffer
409 // capacity (`max_capacity`). That caused the allocated buffer to
410 // remain unfreed when len was 0 (leading to a memory leak).
411 if (p_data != nullptr) {
412 // Call destructors on all active elements to prevent memory leaks
413 cleanup(p_data, 0, len);
414 // free the full allocated buffer
415 deleteArray(p_data, max_capacity);
416 p_data = nullptr;
417 max_capacity = 0;
418 len = 0;
419 }
420 }

435 void resize_internal(int newSize, bool copy, bool shrink = false) {
436 if (newSize <= 0) return;
437 if (newSize > max_capacity || this->p_data == nullptr || shrink) {
438 T *oldData = p_data;
439 int oldBufferLen = this->max_capacity;
440 p_data = newArray(newSize); // new T[newSize+1];
441 assert(p_data != nullptr);
442 this->max_capacity = newSize;
443 if (oldData != nullptr) {
444 if (copy && this->len > 0) {
445 // Element-wise copy using assignment/constructors. Using memmove
446 // here is unsafe for non-trivial types (it copies internal
447 // pointers/ownership and can lead to double-free). Do a proper
448 // element-wise copy so constructors/assignments run.
449 for (int j = 0; j < len; j++) {
450 p_data[j] = oldData[j];
451 }
452 }
453 // Clean up old elements before deallocating, but only those we didn't copy
454 if (!copy && len > 0) {
455 // If we're not copying, destroy all old elements
456 cleanup(oldData, 0, len);
457 }
458 if (shrink) {
459 cleanup(oldData, newSize, oldBufferLen);
460 }
461 deleteArray(oldData, oldBufferLen); // delete [] oldData;
462 }
463 }
464 }
