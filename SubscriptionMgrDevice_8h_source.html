117 Str subscribe(DLNAServiceInfo& service, const char* callbackUrl,
118 const char* sid = nullptr, uint32_t timeoutSec = 1800) {
119 // simple SID generation
120 DlnaLogger.log(DlnaLogLevel::Info, "subscribe: %s %s",
121 StringRegistry::nullStr(service.service_id, "(null)"),
122 StringRegistry::nullStr(callbackUrl, "(null)"));
123
124 bool hasSid = !StrView(sid).isEmpty();
125
126 // NEW subscription without a callback is invalid (GENA requirement)
127 if (!hasSid) {
128 if (StrView(callbackUrl).isEmpty()) {
129 DlnaLogger.log(
130 DlnaLogLevel::Warning,
131 "subscribe: missing CALLBACK header for new subscription");
132 return Str();
133 }
134 }
135
136 // If sid provided, attempt to renew existing subscription for service
137 if (hasSid) {
138 Str renewed = renewSubscription(service, sid, callbackUrl, timeoutSec);
139 if (!renewed.isEmpty()) return renewed;
140 // not found: fall through and create new subscription
141 }
142
143 // generate uuid-based SID
144 char buffer[64];
145 snprintf(buffer, sizeof(buffer), "uuid:%lu", millis());
146
147 // fill subscription
148 Subscription* s = new Subscription();
149 s->sid = buffer;
150 s->callback_url = callbackUrl;
151 s->timeout_sec = timeoutSec;
152 s->seq = 0;
153 s->expires_at = millis() + (uint64_t)timeoutSec * 1000ULL;
154 s->service = &service;
155
156 for (auto& existing_sub : subscriptions) {
157 if (existing_sub->service == &service &&
158 StrView(existing_sub->callback_url).equals(callbackUrl)) {
159 // Found existing subscription for same service and callback URL
160 DlnaLogger.log(DlnaLogLevel::Info,
161 "subscribe: found existing subscription for service "
162 "'%s' and callback '%s', renewing SID '%s'",
163 StringRegistry::nullStr(service.service_id, "(null)"),
164 StringRegistry::nullStr(callbackUrl, "(null)"),
165 existing_sub->sid.c_str());
166 // Renew existing subscription
167 existing_sub->timeout_sec = timeoutSec;
168 existing_sub->expires_at = millis() + (uint64_t)timeoutSec * 1000ULL;
169 return existing_sub->sid;
170 }
171 }
172
173 // add subscription
174 subscriptions.push_back(s);
175 return s->sid;
176 }

190 bool unsubscribe(DLNAServiceInfo& service, const char* sid) {
191 for (size_t i = 0; i < subscriptions.size(); ++i) {
192 Subscription* s = subscriptions[i];
193 if (s->service == &service && StrView(s->sid).equals(sid)) {
194 // remove pending notifications that reference this subscription
195 for (int j = 0; j < pending_list.size();) {
196 if (pending_list[j].p_subscription == s) {
197 pending_list.erase(j);
198 } else {
199 ++j;
200 }
201 }
202 delete s;
203 subscriptions.erase(i);
204 return true;
205 }
206 }
207 return false;
208 }

227 void addChange(DLNAServiceInfo& service,
228 std::function<size_t(Print&, void*)> changeWriter, void* ref) {
229 bool any = false;
230 // do not enqueue if subscriptions are inactive
231 if (!is_active) return;
232
233 // enqueue notifications to be posted later by post()
234 for (int i = 0; i < subscriptions.size(); ++i) {
235 Subscription* sub = subscriptions[i];
236 if (sub->service != &service) continue;
237 any = true;
238 NullPrint np;
239 PendingNotification pn;
240 // store pointer to the subscription entry (no snapshot)
241 pn.p_subscription = subscriptions[i];
242 pn.ref = ref;
243 pn.writer = changeWriter;
244 pn.seq = sub->seq;
245 pending_list.push_back(pn);
246 // increment seq AFTER queuing (so first notification uses SEQ=0)
247 sub->seq++;
248 }
249 if (!any) {
250 DlnaLogger.log(DlnaLogLevel::Info, "service '%s' has no subscriptions",
251 service.service_id);
252 }

264 void publish() {
265 // First remove expired subscriptions so we don't deliver to them.
266 removeExpired();
267 if (pending_list.empty()) return;
268 // Attempt to process each pending notification once. If processing
269 // fails (non-200), leave the entry in the queue for a later retry.
270 int processed = 0;
271 for (int i = 0; i < pending_list.size();) {
272 // copy entry to avoid referencing vector storage while potentially
273 // modifying the vector (erase)
274 PendingNotification pn = pending_list[i];
275
276 // Ensure the subscription pointer is valid; if not, drop the entry.
277 if (pn.p_subscription == nullptr) {
278 DlnaLogger.log(DlnaLogLevel::Warning,
279 "pending notification dropped: missing subscription");
280 pending_list.erase(i);
281 continue;
282 }
283 // determine subscription details
284 Subscription& sub = *pn.p_subscription;
285
286 // convert the sequence number to string
287 char seqBuf[32];
288 snprintf(seqBuf, sizeof(seqBuf), "%d", pn.seq);
289
290 // Build and send HTTP notify as in previous implementation
291 Url cbUrl(sub.callback_url.c_str());
292 DLNAHttpRequest http;
293 http.setHost(cbUrl.host());
294 http.setAgent("tiny-dlna-notify");
295 http.request().put("NT", "upnp:event");
296 http.request().put("NTS", "upnp:propchange");
297 http.request().put("SEQ", seqBuf);
298 http.request().put("SID", sub.sid.c_str());
299 http.setTimeout(DLNA_HTTP_REQUEST_TIMEOUT_MS);
300 DlnaLogger.log(DlnaLogLevel::Info, "Notify %s", cbUrl.url());
301 DlnaLogger.log(DlnaLogLevel::Info, "- SEQ: %s", seqBuf);
302 DlnaLogger.log(DlnaLogLevel::Info, "- SID: %s", sub.sid.c_str());
303 int rc = http.notify(cbUrl, createXML, "text/xml", &pn);
304
305 // log result
306 DlnaLogger.log(DlnaLogLevel::Info, "Notify %s -> %d", cbUrl.url(), rc);
307
308 // remove processed notification on success; otherwise keep it for retry
309 if (rc == 200) {
310 pending_list.erase(i);
311 processed++;
312 // do not increment i: elements shifted down into current index
313 } else {
314 // Increment error count on the stored entry (not the local copy)
315 pending_list[i].error_count++;
316 if (pending_list[i].error_count > MAX_NOTIFY_RETIES) {
317 // give up and drop the entry after too many failed attempts
318 DlnaLogger.log(DlnaLogLevel::Warning,
319 "dropping notify to %s after %d errors with rc=%d %s",
320 cbUrl.url(), pending_list[i].error_count, rc,
321 http.reply().statusMessage());
322 pending_list.erase(i);
323 } else {
324 ++i;
325 }
326 }
327 }
328
329 DlnaLogger.log(
330 DlnaLogLevel::Info,
331 "Published: %d notifications, %d remaining (for %d subscriptions)",
332 processed, pendingCount(), subscriptionsCount());
333 }

342 void removeExpired() {
343 uint64_t now = millis();
344 for (int i = 0; i < subscriptions.size();) {
345 Subscription* s = subscriptions[i];
346 if (s->expires_at != 0 && s->expires_at <= now) {
347 DlnaLogger.log(DlnaLogLevel::Info, "removing expired subscription %s",
348 s->sid.c_str());
349 // Use unsubscribe() which will remove pending notifications and
350 // free the subscription object. Do not increment `i` because the
351 // erase shifts the vector content down into the same index.
352 unsubscribe(*s->service, s->sid.c_str());
353 } else {
354 ++i;
355 }
356 }

410 Str renewSubscription(DLNAServiceInfo& service, const char* sid,
411 const char* callbackUrl, uint32_t timeoutSec) {
412 if (StrView(sid).isEmpty()) return Str();
413 for (int i = 0; i < subscriptions.size(); ++i) {
414 Subscription* ex = subscriptions[i];
415 if (ex->service == &service && StrView(ex->sid).equals(sid)) {
416 // renew: update timeout and expiry, do not change SID
417 ex->timeout_sec = timeoutSec;
418 ex->expires_at = millis() + (uint64_t)timeoutSec * 1000ULL;
419 // if a new callback URL was provided, update it
420 if (callbackUrl != nullptr && !StrView(callbackUrl).isEmpty()) {
421 ex->callback_url = callbackUrl;
422 }
423 DlnaLogger.log(DlnaLogLevel::Info, "renewed subscription %s",
424 ex->sid.c_str());
425 return ex->sid;
426 }
427 }
428 return Str();

452 static size_t createXML(Print& out, void* ref) {
453 EscapingPrint out_esc{out};
454 PendingNotification& pn = *(PendingNotification*)ref;
455 const char* service_abbrev =
456 pn.p_subscription->service->subscription_namespace_abbrev;
457 int instance_id = pn.p_subscription->service->instance_id;
458
459 size_t written = 0;
460 written += out.print("<?xml version=\"1.0\"?>\r\n");
461 written += out.print(
462 "<e:propertyset "
463 "xmlns:e=\"urn:schemas-upnp-org:metadata-1-0/events\">\r\n");
464 written += out.print("<e:property>\r\n");
465 written += out.print("<LastChange>\r\n");
466 // blow should be escaped
467 written +=
468 out_esc.print("<Event xmlns=\"urn:schemas-upnp-org:metadata-1-0/");
469 written += out_esc.print(service_abbrev);
470 written += out_esc.print("/\">\r\n");
471 written += out_esc.print("<InstanceID val=\"");
472 written += out_esc.print(instance_id);
473 written += out_esc.print("\">\r\n");
474 if (pn.writer) {
475 written += pn.writer(out_esc, pn.ref);
476 }
477 written += out_esc.print("</InstanceID>\r\n");
478 written += out_esc.print("</Event>\r\n");
479 // end of escaped
480 written += out.print("</LastChange>\r\n");
481 written += out.print("</e:property>\r\n");
482 written += out.print("</e:propertyset>\r\n");
483 return written;
484 }

497 bool processSubscribeRequest(HttpServer& server, DLNAServiceInfo& service) {
498 // Get headers from request
499 const char* callbackHeader = server.requestHeader().get("CALLBACK");
500 const char* timeoutHeader = server.requestHeader().get("TIMEOUT");
501 const char* sidHeader = server.requestHeader().get("SID");
502
503 // Log the incoming headers
504 DlnaLogger.log(DlnaLogLevel::Info, "- SUBSCRIBE CALLBACK: %s",
505 callbackHeader ? callbackHeader : "null");
506 DlnaLogger.log(DlnaLogLevel::Info, "- SUBSCRIBE TIMEOUT: %s",
507 timeoutHeader ? timeoutHeader : "null");
508 DlnaLogger.log(DlnaLogLevel::Info, "- SUBSCRIBE SID: %s",
509 sidHeader ? sidHeader : "null");
510
511 // Parse and clean callback URL (remove angle brackets)
512 Str cbStr;
513 if (callbackHeader) {
514 cbStr = callbackHeader;
515 cbStr.replace("<", "");
516 cbStr.replace(">", "");
517 cbStr.trim();
518 }
519
520 // Sanitize SID (remove optional angle brackets)
521 Str sidStr;
522 if (sidHeader) {
523 sidStr = sidHeader;
524 sidStr.replace("<", "");
525 sidStr.replace(">", "");
526 sidStr.trim();
527 }
528
529 // Parse timeout (extract seconds from "Second-1800" format)
530 uint32_t tsec = 1800;
531 if (timeoutHeader && StrView(timeoutHeader).startsWith("Second-")) {
532 tsec = atoi(timeoutHeader + 7);
533 }
534
535 // Create or renew subscription
536 const char* callbackPtr = cbStr.isEmpty() ? nullptr : cbStr.c_str();
537 const char* sidPtr = sidStr.isEmpty() ? nullptr : sidStr.c_str();
538
539 Str sid = subscribe(service, callbackPtr, sidPtr, tsec);
540 if (sid.isEmpty()) {
541 DlnaLogger.log(DlnaLogLevel::Warning,
542 "subscribe request rejected (missing data)");
543 server.replyHeader().setValues(412, "Precondition Failed");
544 server.replyHeader().put("Content-Length", 0);
545 server.replyHeader().write(server.client());
546 server.endClient();
547 return false;
548 }
549 DlnaLogger.log(DlnaLogLevel::Info, "- SID: %s", sid.c_str());
550
551 // Send SUBSCRIBE response
552 server.replyHeader().setValues(200, "OK");
553 server.replyHeader().put("SID", sid.c_str());
554 server.replyHeader().put("TIMEOUT", "Second-1800");
555 server.replyHeader().put("Content-Length", 0);
556 server.replyHeader().write(server.client());
557 server.endClient();
558
559 return true;
560 }

573 bool processUnsubscribeRequest(HttpServer& server, DLNAServiceInfo& service) {
574 const char* sid = server.requestHeader().get("SID");
575
576 DlnaLogger.log(DlnaLogLevel::Info, "- UNSUBSCRIBE SID: %s",
577 sid ? sid : "null");
578
579 if (sid) {
580 bool ok = unsubscribe(service, sid);
581 if (ok) {
582 DlnaLogger.log(DlnaLogLevel::Info, "Unsubscribed: %s", sid);
583 server.replyOK();
584 return true;
585 }
586 }
587
588 // SID not found or invalid
589 DlnaLogger.log(DlnaLogLevel::Warning,
590 "handleUnsubscribeRequest: failed for SID %s",
591 sid ? sid : "(null)");
592 server.replyNotFound();
593 return false;
594 }
