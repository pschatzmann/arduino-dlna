112 Str subscribe(DLNAServiceInfo& service, const char* callbackUrl,
113 const char* sid = nullptr, uint32_t timeoutSec = 1800) {
114 // simple SID generation
115 DlnaLogger.log(DlnaLogLevel::Info, "subscribe: %s %s",
116 StringRegistry::nullStr(service.service_id, "(null)"),
117 StringRegistry::nullStr(callbackUrl, "(null)"));
118 // If sid provided, attempt to renew existing subscription for service
119 if (sid != nullptr && !StrView(sid).isEmpty()) {
120 Str renewed = renewSubscription(service, sid, callbackUrl, timeoutSec);
121 if (!renewed.isEmpty()) return renewed;
122 // not found: fall through and create new subscription
123 }
124
125 // generate uuid-based SID
126 char buffer[64];
127 snprintf(buffer, sizeof(buffer), "uuid:%lu", millis());
128
129 // fill subscription
130 Subscription* s = new Subscription();
131 s->sid = buffer;
132 s->callback_url = callbackUrl;
133 s->timeout_sec = timeoutSec;
134 s->seq = 0;
135 s->expires_at = millis() + (uint64_t)timeoutSec * 1000ULL;
136 s->service = &service;
137
138 // add subscription
139 subscriptions.push_back(s);
140 return s->sid;
141 }

156 for (size_t i = 0; i < subscriptions.size(); ++i) {

227 void publish() {
228 // First remove expired subscriptions so we don't deliver to them.
229 removeExpired();
230 if (pending_list.empty()) return;
231 // Attempt to process each pending notification once. If processing
232 // fails (non-200), leave the entry in the queue for a later retry.
233 int processed = 0;
234 for (int i = 0; i < pending_list.size();) {
235 // copy entry to avoid referencing vector storage while potentially
236 // modifying the vector (erase)
237 PendingNotification pn = pending_list[i];
238
239 // Ensure the subscription pointer is valid; if not, drop the entry.
240 if (pn.p_subscription == nullptr) {
241 DlnaLogger.log(DlnaLogLevel::Warning,
242 "pending notification dropped: missing subscription");
243 pending_list.erase(i);
244 continue;
245 }
246
247 Subscription& sub = *pn.p_subscription;
248
249 // Build and send HTTP notify as in previous implementation
250 Url cbUrl(sub.callback_url.c_str());
251 //Url cbUrl("http://192.168.1.44:8000");
252 NullPrint np;
253 DLNAHttpRequest http;
254 http.setHost(cbUrl.host());
255 http.setAgent("tiny-dlna-notify");
256 http.request().put("NT", "upnp:event");
257 http.request().put("NTS", "upnp:propchange");
258 char seqBuf[32];
259 snprintf(seqBuf, sizeof(seqBuf), "%d", sub.seq);
260 http.request().put("SEQ", seqBuf);
261 http.request().put("SID", sub.sid.c_str());
262 http.setTimeout(DLNA_HTTP_REQUEST_TIMEOUT_MS);
263 // pass the pending notification as reference
264 //int len = createXML(Serial, &pn);
265 int len = createXML(np, &pn);
266 int rc = http.notify(cbUrl, len, createXML, "text/xml", &pn);
267
268 DlnaLogger.log(DlnaLogLevel::Info, "Notify %s -> %d", cbUrl.url(), rc);
269 http.stop();
270
271 // remove processed notification on success; otherwise keep it for retry
272 if (rc == 200) {
273 pending_list.erase(i);
274 processed++;
275 // do not increment i: elements shifted down into current index
276 } else {
277 // Increment error count on the stored entry (not the local copy)
278 pending_list[i].error_count++;
279 if (pending_list[i].error_count > MAX_NOTIFY_RETIES) {
280 // give up and drop the entry after too many failed attempts
281 DlnaLogger.log(DlnaLogLevel::Warning,
282 "dropping notify to %s after %d errors with rc=%d %s", cbUrl.url(),
283 pending_list[i].error_count, rc, http.reply().statusMessage());
284 pending_list.erase(i);
285 } else {
286 ++i;
287 }
288 }
289 }
290
291 DlnaLogger.log(
292 DlnaLogLevel::Info,
293 "Published: %d notifications, %d remaining (for %d subscriptions)",
294 processed, pendingCount(), subscriptionsCount());
295 }

304 void removeExpired() {
305 uint64_t now = millis();
306 for (int i = 0; i < subscriptions.size();) {
307 Subscription* s = subscriptions[i];
308 if (s->expires_at != 0 && s->expires_at <= now) {
309 DlnaLogger.log(DlnaLogLevel::Info, "removing expired subscription %s",
310 s->sid.c_str());
311 // Use unsubscribe() which will remove pending notifications and
312 // free the subscription object. Do not increment `i` because the
313 // erase shifts the vector content down into the same index.
314 unsubscribe(*s->service, s->sid.c_str());
315 } else {
316 ++i;
317 }
318 }

374 Str renewSubscription(DLNAServiceInfo& service, const char* sid,
375 const char* callbackUrl, uint32_t timeoutSec) {
376 if (sid == nullptr || StrView(sid).isEmpty()) return Str();
377 for (int i = 0; i < subscriptions.size(); ++i) {
378 Subscription* ex = subscriptions[i];
379 if (ex->service == &service && StrView(ex->sid).equals(sid)) {
380 // renew: update timeout and expiry, do not change SID
381 ex->timeout_sec = timeoutSec;
382 ex->expires_at = millis() + (uint64_t)timeoutSec * 1000ULL;
383 // if a new callback URL was provided, update it
384 if (callbackUrl != nullptr && !StrView(callbackUrl).isEmpty()) {
385 ex->callback_url = callbackUrl;
386 }
387 DlnaLogger.log(DlnaLogLevel::Info, "renewed subscription %s",
388 ex->sid.c_str());
389 return ex->sid;
390 }
391 }
392 return Str();

417 static size_t createXML(Print& out, void* ref) {
418 EscapingPrint out_esc{out};
419 PendingNotification& pn = *(PendingNotification*)ref;
420 const char* service_abbrev = pn.p_subscription->service->subscription_namespace_abbrev;
421 int instance_id = pn.p_subscription->service->instance_id;
422
423
424 size_t written = 0;
425 written += out.print("<?xml version=\"1.0\"?>\r\n");
426 written += out.print(
427 "<e:propertyset "
428 "xmlns:e=\"urn:schemas-upnp-org:metadata-1-0/events\">\r\n");
429 written += out.print("<e:property>\r\n");
430 written += out.print("<LastChange>\r\n");
431 // blow should be escaped
432 written += out_esc.print("<Event xmlns=\"urn:schemas-upnp-org:metadata-1-0/");
433 written += out_esc.print(service_abbrev);
434 written += out_esc.print("/\">\r\n");
435 written += out_esc.print("<InstanceID val=\"");
436 written += out_esc.print(instance_id);
437 written += out_esc.print("\">\r\n");
438 if (pn.writer) {
439 written += pn.writer(out_esc, pn.ref);
440 }
441 written += out_esc.print("</InstanceID>\r\n");
442 written += out_esc.print("</Event>\r\n");
443 // end of escaped
444 written += out.print("</LastChange>\r\n");
445 written += out.print("</e:property>\r\n");
446 written += out.print("</e:propertyset>\r\n");
