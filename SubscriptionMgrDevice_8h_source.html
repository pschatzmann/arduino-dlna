36struct PendingNotification {
37 // Pointer to the original subscription entry at enqueue time. We keep
38 // a pointer only (no copy) to reduce memory overhead. Note: this may
39 // become invalid if the subscriptions vector is reallocated/modified,
40 // callers must handle null/dangling pointers conservatively.
41 Subscription* p_subscription = nullptr;

43 int error_count = 0;

99 bool unsubscribe(DLNAServiceInfo& service, const char* sid) {
100 for (size_t i = 0; i < subscriptions.size(); ++i) {
101 Subscription* s = subscriptions[i];
102 if (s->service == &service && StrView(s->sid).equals(sid)) {
103 // remove pending notifications that reference this subscription
104 for (int j = 0; j < pending_list.size();) {
105 if (pending_list[j].p_subscription == s) {
106 pending_list.erase(j);
107 } else {
108 ++j;
109 }
110 }
111 delete s;
112 subscriptions.erase(i);
113 return true;
114 }
115 }
116 return false;

120 void addChange(DLNAServiceInfo& service, const char* xmlTag) {
121 bool any = false;
122 if (StrView(xmlTag).isEmpty()) {
123 DlnaLogger.log(DlnaLogLevel::Warning, "empty xmlTag for service '%s'",
124 service.service_id);
125 return;
126 }
127 // enqueue notifications to be posted later by post()
128 for (int i = 0; i < subscriptions.size(); ++i) {
129 Subscription* sub = subscriptions[i];
130 if (sub->service != &service) continue;
131 any = true;
132 // increment seq now (reserve sequence number)
133 sub->seq++;
134 PendingNotification pn;
135 // store pointer to the subscription entry (no snapshot)
136 pn.p_subscription = subscriptions[i];
137 pn.changeNode = xmlTag;
138 pending_list.push_back(pn);
139 }
140 if (!any) {
141 DlnaLogger.log(DlnaLogLevel::Error, "service '%s' not found",
142 service.service_id);

148 void publish() {
149 // First remove expired subscriptions so we don't deliver to them.
150 removeExpired();
151 if (pending_list.empty()) return;
152 // Attempt to process each pending notification once. If processing
153 // fails (non-200), leave the entry in the queue for a later retry.
154 int processed = 0;
155 for (int i = 0; i < pending_list.size();) {
156 // copy entry to avoid referencing vector storage while potentially
157 // modifying the vector (erase)
158 PendingNotification pn = pending_list[i];
159
160 // Ensure the subscription pointer is valid; if not, drop the entry.
161 if (pn.p_subscription == nullptr) {
162 DlnaLogger.log(DlnaLogLevel::Warning,
163 "pending notification dropped: missing subscription");
164 pending_list.erase(i);
165 continue;
166 }
167
168 Subscription& sub = *pn.p_subscription;
169
170 // Build and send HTTP notify as in previous implementation
171 Url cbUrl(sub.callback_url.c_str());
172 DLNAHttpRequest http;
173 http.setHost(cbUrl.host());
174 http.setAgent("tiny-dlna-notify");
175 http.request().put("NT", "upnp:event");
176 http.request().put("NTS", "upnp:propchange");
177 char seqBuf[32];
178 snprintf(seqBuf, sizeof(seqBuf), "%d", sub.seq);
179 http.request().put("SEQ", seqBuf);
180 http.request().put("SID", sub.sid.c_str());
181
182 // compute xml length
183 NullPrint np;
184 size_t xmlLen = 0;
185 const char* serviceAbbrev =
186 sub.service ? sub.service->subscription_namespace_abbrev : "";
187 if (is_log_xml)
188 xmlLen = createXML(Serial, serviceAbbrev, pn.changeNode.c_str());
189 else
190 xmlLen = createXML(np, serviceAbbrev, pn.changeNode.c_str());
191
192 auto printXML = [this, serviceAbbrev, pn](Print& out) {
193 createXML(out, serviceAbbrev, pn.changeNode.c_str());
194 };
195
196 // post the notification
197 int rc = http.post(cbUrl, xmlLen, printXML, "text/xml");
198 DlnaLogger.log(DlnaLogLevel::Info, "Notify %s -> %d", cbUrl.url(), rc);
199
200 // remove processed notification on success; otherwise keep it for retry
201 if (rc == 200) {
202 pending_list.erase(i);
203 processed++;
204 // do not increment i: elements shifted down into current index
205 } else {
206 // Increment error count on the stored entry (not the local copy)
207 // increment the stored entry's error count
208 pending_list[i].error_count++;
209 if (pending_list[i].error_count > MAX_SEND_ERRORS) {
210 // give up and drop the entry after too many failed attempts
211 DlnaLogger.log(DlnaLogLevel::Warning,
212 "dropping notify to %s after %d errors", cbUrl.url(),
213 pending_list[i].error_count);
214 pending_list.erase(i);
215 } else {
216 ++i;
217 }
218 }
219 }
220
221 DlnaLogger.log(
222 DlnaLogLevel::Info,
223 "Published: %d notifications, %d remaining (for %d subscriptions)",
224 processed, pendingCount(), subscriptionsCount());
225 }

230 void removeExpired() {
231 uint64_t now = millis();
232 for (int i = 0; i < subscriptions.size();) {
233 Subscription* s = subscriptions[i];
234 if (s->expires_at != 0 && s->expires_at <= now) {
235 DlnaLogger.log(DlnaLogLevel::Info, "removing expired subscription %s",
236 s->sid.c_str());
237 // Use unsubscribe() which will remove pending notifications and
238 // free the subscription object. Do not increment `i` because the
239 // erase shifts the vector content down into the same index.
240 unsubscribe(*s->service, s->sid.c_str());
241 } else {
242 ++i;
243 }
244 }
