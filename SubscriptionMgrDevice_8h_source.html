129 Str subscribe(DLNAServiceInfo& service, const char* callbackUrl,
130 const char* sid = nullptr,
131 uint32_t timeoutSec = 1800) override {
132 // simple SID generation
133 DlnaLogger.log(DlnaLogLevel::Info, "subscribe: %s %s",
134 StrView(service.service_id).c_str(),
135 StrView(callbackUrl).c_str());
136
137 bool hasSid = !StrView(sid).isEmpty();
138
139 // NEW subscription without a callback is invalid (GENA requirement)
140 if (!hasSid) {
141 if (StrView(callbackUrl).isEmpty()) {
142 DlnaLogger.log(
143 DlnaLogLevel::Warning,
144 "subscribe: missing CALLBACK header for new subscription");
145 return Str();
146 }
147 }
148
149 // If sid provided, attempt to renew existing subscription for service
150 if (hasSid) {
151 Str renewed = renewSubscription(service, sid, callbackUrl, timeoutSec);
152 if (!renewed.isEmpty()) return renewed;
153 // not found: fall through and create new subscription
154 }
155
156 // generate uuid-based SID
157 char buffer[64];
158 snprintf(buffer, sizeof(buffer), "uuid:%lu", millis());
159
160 // fill subscription
161 Subscription* s = new Subscription();
162 s->sid = buffer;
163 s->callback_url = callbackUrl;
164 s->timeout_sec = timeoutSec;
165 s->seq = 0;
166 s->expires_at = millis() + (uint64_t)timeoutSec * 1000ULL;
167 s->service = &service;
168
169 for (auto& existing_sub : subscriptions) {
170 if (existing_sub->service == &service &&
171 StrView(existing_sub->callback_url).equals(callbackUrl)) {
172 // Found existing subscription for same service and callback URL
173 DlnaLogger.log(DlnaLogLevel::Info,
174 "subscribe: found existing subscription for service "
175 "'%s' and callback '%s', renewing SID '%s'",
176 StrView(service.service_id).c_str(),
177 StrView(callbackUrl).c_str(),
178 existing_sub->sid.c_str());
179 // Renew existing subscription
180 existing_sub->timeout_sec = timeoutSec;
181 existing_sub->expires_at = millis() + (uint64_t)timeoutSec * 1000ULL;
182 return existing_sub->sid;
183 }
184 }
185
186 // add subscription
187 subscriptions.push_back(s);
188 return s->sid;
189 }

203 bool unsubscribe(DLNAServiceInfo& service, const char* sid) override {
204 for (size_t i = 0; i < subscriptions.size(); ++i) {
205 Subscription* s = subscriptions[i];
206 if (s->service == &service && StrView(s->sid).equals(sid)) {
207 // remove pending notifications that reference this subscription
208 for (int j = 0; j < pending_list.size();) {
209 if (pending_list[j].p_subscription == s) {
210 pending_list.erase(j);
211 } else {
212 ++j;
213 }
214 }
215 delete s;
216 subscriptions.erase(i);
217 return true;
218 }
219 }
220 return false;

240 void addChange(DLNAServiceInfo& service,
241 std::function<size_t(Print&, void*)> changeWriter,
242 void* ref) override {
243 bool any = false;
244 // do not enqueue if subscriptions are inactive
245 if (!is_active) return;
246
247 // enqueue notifications to be posted later by post()
248 for (int i = 0; i < subscriptions.size(); ++i) {
249 Subscription* sub = subscriptions[i];
250 if (sub->service != &service) continue;
251 any = true;
252 NullPrint np;
253 PendingNotification pn;
254 // store pointer to the subscription entry (no snapshot)
255 pn.p_subscription = subscriptions[i];
256 pn.ref = ref;
257 pn.writer = changeWriter;
258 pn.seq = sub->seq;
259 pending_list.push_back(pn);
260 // increment seq AFTER queuing (so first notification uses SEQ=0)
261 sub->seq++;
262 }
263 if (!any) {
264 DlnaLogger.log(DlnaLogLevel::Info, "service '%s' has no subscriptions",
265 service.service_id);
266 }

278 int publish() override {
279 // First remove expired subscriptions so we don't deliver to them.
280 removeExpired();
281 if (pending_list.empty()) return 0;
282 // Attempt to process each pending notification once. If processing
283 // fails (non-200), leave the entry in the queue for a later retry.
284 int processed = 0;
285 for (int i = 0; i < pending_list.size();) {
286 // copy entry to avoid referencing vector storage while potentially
287 // modifying the vector (erase)
288 PendingNotification pn = pending_list[i];
289
290 // Ensure the subscription pointer is valid; if not, drop the entry.
291 if (pn.p_subscription == nullptr) {
292 DlnaLogger.log(DlnaLogLevel::Warning,
293 "pending notification dropped: missing subscription");
294 pending_list.erase(i);
295 continue;
296 }
297 // determine subscription details
298 Subscription& sub = *pn.p_subscription;
299
300 // convert the sequence number to string
301 char seqBuf[32];
302 snprintf(seqBuf, sizeof(seqBuf), "%d", pn.seq);
303
304 // Build and send HTTP notify as in previous implementation
305 Url cbUrl(sub.callback_url.c_str());
306 HttpRequest<ClientType> http;
307 http.setTimeout(DLNA_HTTP_REQUEST_TIMEOUT_MS);
308 http.setHost(cbUrl.host());
309 http.setAgent("tiny-dlna-notify");
310 http.request().put("NT", "upnp:event");
311 http.request().put("NTS", "upnp:propchange");
312 http.request().put("SEQ", seqBuf);
313 http.request().put("SID", sub.sid.c_str());
314 DlnaLogger.log(DlnaLogLevel::Info, "Notify %s", cbUrl.url());
315 DlnaLogger.log(DlnaLogLevel::Info, "- SEQ: %s", seqBuf);
316 DlnaLogger.log(DlnaLogLevel::Info, "- SID: %s", sub.sid.c_str());
317 int rc = http.notify(cbUrl, createXML, "text/xml", &pn);
318
319 // log result
320 DlnaLogger.log(DlnaLogLevel::Info, "Notify %s -> %d", cbUrl.url(), rc);
321
322 // remove processed notification on success; otherwise keep it for retry
323 if (rc == 200) {
324 pending_list.erase(i);
325 processed++;
326 // do not increment i: elements shifted down into current index
327 } else {
328 // Increment error count on the stored entry (not the local copy)
329 pending_list[i].error_count++;
330 if (pending_list[i].error_count > MAX_NOTIFY_RETIES) {
331 // give up and drop the entry after too many failed attempts
332 DlnaLogger.log(DlnaLogLevel::Warning,
333 "dropping notify to %s after %d errors with rc=%d %s",
334 cbUrl.url(), pending_list[i].error_count, rc,
335 http.reply().statusMessage());
336 pending_list.erase(i);
337 } else {
338 ++i;
339 }
340 }
341 }
342
343 DlnaLogger.log(
344 DlnaLogLevel::Info,
345 "Published: %d notifications, %d remaining (for %d subscriptions)",
346 processed, pendingCount(), subscriptionsCount());
347 return processed;
348 }

357 void removeExpired() override {
358 uint64_t now = millis();
359 for (int i = 0; i < subscriptions.size();) {
360 Subscription* s = subscriptions[i];
361 if (s->expires_at != 0 && s->expires_at <= now) {
362 DlnaLogger.log(DlnaLogLevel::Info, "removing expired subscription %s",
363 s->sid.c_str());
364 // Use unsubscribe() which will remove pending notifications and
365 // free the subscription object. Do not increment `i` because the
366 // erase shifts the vector content down into the same index.
367 unsubscribe(*s->service, s->sid.c_str());
368 } else {
369 ++i;
370 }
371 }
372 }

426 Str renewSubscription(DLNAServiceInfo& service, const char* sid,
427 const char* callbackUrl, uint32_t timeoutSec) {
428 if (StrView(sid).isEmpty()) return Str();
429 for (int i = 0; i < subscriptions.size(); ++i) {
430 Subscription* ex = subscriptions[i];
431 if (ex->service == &service && StrView(ex->sid).equals(sid)) {
432 // renew: update timeout and expiry, do not change SID
433 ex->timeout_sec = timeoutSec;
434 ex->expires_at = millis() + (uint64_t)timeoutSec * 1000ULL;
435 // if a new callback URL was provided, update it
436 if (callbackUrl != nullptr && !StrView(callbackUrl).isEmpty()) {
437 ex->callback_url = callbackUrl;
438 }
439 DlnaLogger.log(DlnaLogLevel::Info, "renewed subscription %s",
440 ex->sid.c_str());
441 return ex->sid;
442 }
443 }
444 return Str();

468 static size_t createXML(Print& out, void* ref) {
469 EscapingPrint out_esc{out};
470 PendingNotification& pn = *(PendingNotification*)ref;
471 const char* service_abbrev =
472 pn.p_subscription->service->subscription_namespace_abbrev;
473 int instance_id = pn.p_subscription->service->instance_id;
474
475 size_t written = 0;
476 written += out.print("<?xml version=\"1.0\"?>\r\n");
477 written += out.print(
478 "<e:propertyset "
479 "xmlns:e=\"urn:schemas-upnp-org:metadata-1-0/events\">\r\n");
480 written += out.print("<e:property>\r\n");
481 written += out.print("<LastChange>\r\n");
482 // blow should be escaped
483 written +=
484 out_esc.print("<Event xmlns=\"urn:schemas-upnp-org:metadata-1-0/");
485 written += out_esc.print(service_abbrev);
486 written += out_esc.print("/\">\r\n");
487 written += out_esc.print("<InstanceID val=\"");
488 written += out_esc.print(instance_id);
489 written += out_esc.print("\">\r\n");
490 if (pn.writer) {
491 written += pn.writer(out_esc, pn.ref);
492 }
493 written += out_esc.print("</InstanceID>\r\n");
494 written += out_esc.print("</Event>\r\n");
495 // end of escaped
496 written += out.print("</LastChange>\r\n");
497 written += out.print("</e:property>\r\n");
498 written += out.print("</e:propertyset>\r\n");
499 return written;

513 bool processSubscribeRequest(IHttpServer& server,
514 DLNAServiceInfo& service) override {
515 // Get headers from request
516 const char* callbackHeader = server.requestHeader().get("CALLBACK");
517 const char* timeoutHeader = server.requestHeader().get("TIMEOUT");
518 const char* sidHeader = server.requestHeader().get("SID");
519
520 // Log the incoming headers
521 DlnaLogger.log(DlnaLogLevel::Info, "- SUBSCRIBE CALLBACK: %s",
522 callbackHeader ? callbackHeader : "null");
523 DlnaLogger.log(DlnaLogLevel::Info, "- SUBSCRIBE TIMEOUT: %s",
524 timeoutHeader ? timeoutHeader : "null");
525 DlnaLogger.log(DlnaLogLevel::Info, "- SUBSCRIBE SID: %s",
526 sidHeader ? sidHeader : "null");
527
528 // Parse and clean callback URL (remove angle brackets)
529 Str cbStr;
530 if (callbackHeader) {
531 cbStr = callbackHeader;
532 cbStr.replace("<", "");
533 cbStr.replace(">", "");
534 cbStr.trim();
535 }
536
537 // Sanitize SID (remove optional angle brackets)
538 Str sidStr;
539 if (sidHeader) {
540 sidStr = sidHeader;
541 sidStr.replace("<", "");
542 sidStr.replace(">", "");
543 sidStr.trim();
544 }
545
546 // Parse timeout (extract seconds from "Second-1800" format)
547 uint32_t tsec = 1800;
548 if (timeoutHeader && StrView(timeoutHeader).startsWith("Second-")) {
549 tsec = atoi(timeoutHeader + 7);
550 }
551
552 // Create or renew subscription
553 const char* callbackPtr = cbStr.isEmpty() ? nullptr : cbStr.c_str();
554 const char* sidPtr = sidStr.isEmpty() ? nullptr : sidStr.c_str();
555
556 Str sid = subscribe(service, callbackPtr, sidPtr, tsec);
557 if (sid.isEmpty()) {
558 DlnaLogger.log(DlnaLogLevel::Warning,
559 "subscribe request rejected (missing data)");
560 server.replyHeader().setValues(412, "Precondition Failed");
561 server.replyHeader().put("Content-Length", 0);
562 server.replyHeader().write(server.client());
563 server.endClient();
564 return false;
565 }
566 DlnaLogger.log(DlnaLogLevel::Info, "- SID: %s", sid.c_str());
567
568 // Send SUBSCRIBE response
569 server.replyHeader().setValues(200, "OK");
570 server.replyHeader().put("SID", sid.c_str());
571 server.replyHeader().put("TIMEOUT", "Second-1800");
572 server.replyHeader().put("Content-Length", 0);
573 server.replyHeader().write(server.client());
574 server.endClient();
575
576 return true;
577 }

590 bool processUnsubscribeRequest(IHttpServer& server,
591 DLNAServiceInfo& service) override {
592 const char* sid = server.requestHeader().get("SID");
593
594 DlnaLogger.log(DlnaLogLevel::Info, "- UNSUBSCRIBE SID: %s",
595 sid ? sid : "null");
596
597 if (sid) {
598 bool ok = unsubscribe(service, sid);
599 if (ok) {
600 DlnaLogger.log(DlnaLogLevel::Info, "Unsubscribed: %s", sid);
601 server.replyOK();
602 return true;
603 }
604 }
605
606 // SID not found or invalid
607 DlnaLogger.log(DlnaLogLevel::Warning,
608 "handleUnsubscribeRequest: failed for SID %s",
609 sid ? sid : "(null)");
610 server.replyNotFound();
611 return false;
612 }
