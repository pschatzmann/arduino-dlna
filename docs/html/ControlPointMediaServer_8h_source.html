214  void processResult(const char* resultXml, ItemCallback itemCallback) {
215  if (!resultXml) return;
216  StrView res(resultXml);
217  int pos = 0;
218  while (true) {
219  int itPos = res.indexOf("<item", pos);
220  if (itPos < 0) break;
221  int itEnd = res.indexOf("</item>", itPos);
222  if (itEnd < 0) break;
223  int headerEnd = res.indexOf('>', itPos);
224  if (headerEnd < 0 || headerEnd >= itEnd) break;
225 
226  tiny_dlna::MediaItem item;
227  // extract id attribute
228  int idPos = res.indexOf("id=\"", itPos);
229  if (idPos >= 0 && idPos < headerEnd) {
230  int valStart = idPos + 4;
231  int valEnd = res.indexOf('"', valStart);
232  if (valEnd > valStart) {
233  Str tmp;
234  tmp.copyFrom(res.c_str() + valStart, valEnd - valStart);
235  item.id = strings.add((char*)tmp.c_str());
236  }
237  }
238 
239  // title
240  int t1 = res.indexOf("<dc:title>", headerEnd);
241  if (t1 >= 0 && t1 < itEnd) {
242  int t1s = t1 + strlen("<dc:title>");
243  int t1e = res.indexOf("</dc:title>", t1s);
244  if (t1e > t1s) {
245  Str tmp;
246  tmp.copyFrom(res.c_str() + t1s, t1e - t1s);
247  item.title = strings.add((char*)tmp.c_str());
248  }
249  }

253  if (r1 >= 0 && r1 < itEnd) {
254  int r1s = res.indexOf('>', r1);
255  if (r1s >= 0 && r1s < itEnd) {

261  item.res = strings.add((char*)tmp.c_str());
262  }
263  }
264  }
