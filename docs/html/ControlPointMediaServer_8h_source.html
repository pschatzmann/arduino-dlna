193  // Parse DIDL-Lite Result and invoke callback for each item
194  void processResult(const char* resultXml, ItemCallback itemCallback) {
195  if (!resultXml) return;
196  StrView res(resultXml);
197  int pos = 0;
198  while (true) {
199  int itPos = res.indexOf("<item", pos);
200  if (itPos < 0) break;
201  int itEnd = res.indexOf("</item>", itPos);
202  if (itEnd < 0) break;
203  int headerEnd = res.indexOf('>', itPos);
204  if (headerEnd < 0 || headerEnd >= itEnd) break;
205 
206  MediaServer::MediaItem item;
207  // extract id attribute
208  int idPos = res.indexOf("id=\"", itPos);
209  if (idPos >= 0 && idPos < headerEnd) {
210  int valStart = idPos + 4;
211  int valEnd = res.indexOf('"', valStart);
212  if (valEnd > valStart) {
213  Str tmp;
214  tmp.copyFrom(res.c_str() + valStart, valEnd - valStart);
215  item.id = strings.add((char*)tmp.c_str());
216  }
217  }
218 
219  // title
220  int t1 = res.indexOf("<dc:title>", headerEnd);
221  if (t1 >= 0 && t1 < itEnd) {
222  int t1s = t1 + strlen("<dc:title>");
223  int t1e = res.indexOf("</dc:title>", t1s);
224  if (t1e > t1s) {
225  Str tmp;
226  tmp.copyFrom(res.c_str() + t1s, t1e - t1s);
227  item.title = strings.add((char*)tmp.c_str());
228  }
229  }
230 
231  // res
232  int r1 = res.indexOf("<res", headerEnd);
233  if (r1 >= 0 && r1 < itEnd) {
234  int r1s = res.indexOf('>', r1);
235  if (r1s >= 0 && r1s < itEnd) {
236  r1s += 1;
237  int r1e = res.indexOf("</res>", r1s);
238  if (r1e > r1s) {
239  Str tmp;
240  tmp.copyFrom(res.c_str() + r1s, r1e - r1s);
241  item.res = strings.add((char*)tmp.c_str());
242  }
243  }
244  }
245 
246  if (itemCallback) itemCallback(item, reference);
247  pos = itEnd + 7;
248  }
249  }
