200  // Run the parser. This is a small, forgiving parser suitable for the
201  // embedded use-cases in this project (DIDL fragments, simple SCPD parsing).
202  // It is not a full XML validator but handles start/end tags, comments,
203  // processing instructions and self-closing tags. For each text node that
204  // contains non-whitespace characters the provided callback is invoked with
205  // the text and the current element path.
206  void do_parse() {
207  const char* s = str_view.c_str();
208  int len = str_view.length();
209  int pos = 0;
210 
211  while (pos < len) {
212  int lt = str_view.indexOf('<', pos);
213  if (lt < 0) break;
214 
215  // Handle text between pos and lt
216  if (lt > pos) {
217  int ts = pos;
218  int te = lt;
219  while (ts < te && isspace((unsigned char)s[ts])) ts++;
220  while (te > ts && isspace((unsigned char)s[te - 1])) te--;
221  emitTextSegment(s, ts, te);
222  }
223 
224  // comment or processing instruction handling
225  int newPos = handleCommentOrPI(lt, s, len);
226  if (newPos != lt) {
227  pos = newPos;
228  continue;
229  }
230 
231  // find closing '>' (respect quotes)
232  int gt = findGt(s, lt, len);
233  if (gt < 0) break; // malformed
234 
235  // end tag
236  if (lt + 1 < len && s[lt + 1] == '/') {
237  pos = handleEndTag(s, lt, gt);
238  continue;
239  }
240 
241  // start tag (or self-closing)
242  handleStartTag(s, lt, gt);
243 
244  // callback for the tag itself
245  emitTagSegment(s, lt, gt);
246 
247  // detect self-closing and pop if needed
248  bool selfClosing = false;
249  int back = gt - 1;
250  while (back > lt && isspace((unsigned char)s[back])) back--;
251  if (back > lt && s[back] == '/') selfClosing = true;
252 
253  pos = gt + 1;
254 
255  if (selfClosing && path.size() > 0) {
256  path.erase(path.size() - 1);
257  if (contentStarts.size() > 0)
258  contentStarts.erase(contentStarts.size() - 1);
259  }
260  }
261 
262  // trailing text
263  int firstLt = str_view.indexOf('<', 0);
264  if (firstLt < 0) firstLt = 0;
265  if (firstLt < len) handleTrailingText(s, len);
266  }
