407  str_print.write(buffer, len);

410  // log result
411  DlnaLogger.log(DlnaLogLevel::Debug, str_print.c_str());

414  // Try to parse SOAP response body and extract return arguments
415  // e.g. <u:SomeActionResponse> <TrackDuration>01:23:45</TrackDuration> ...</u:SomeActionResponse>
416  const char* resp = str_print.c_str();
417  if (resp != nullptr && *resp != '\0') {
418  StrView soap(resp);
419 
420  // Build response tag name: ActionNameResponse
421  char respTagBuf[200] = {0};
422  snprintf(respTagBuf, sizeof(respTagBuf), "%sResponse", action.action);
423  int respPos = soap.indexOf(respTagBuf);
424  if (respPos >= 0) {
425  // find opening '<' before respPos
426  int openPos = respPos;
427  while (openPos > 0 && soap[openPos] != '<') openPos--;
428  if (openPos < 0) openPos = 0;
429  int openEnd = soap.indexOf('>', openPos);
430  if (openEnd >= 0) {
431  int payloadStart = openEnd + 1;
432  // find end tag
433  char endTagBuf[220] = {0};
434  snprintf(endTagBuf, sizeof(endTagBuf), "</%sResponse>", action.action);
435  int payloadEnd = soap.indexOf(endTagBuf, payloadStart);
436  if (payloadEnd < 0) payloadEnd = soap.length();
437 
438  int pos = payloadStart;
439  // iterate over child elements inside the response
440  while (true) {
441  int lt = soap.indexOf('<', pos);
442  if (lt < 0 || lt >= payloadEnd) break;
443  // skip closing tags
444  if (lt + 1 < soap.length() && soap[lt + 1] == '/') {
445  pos = lt + 2;
446  continue;
447  }
448  int nameStart = lt + 1;
449  // read element name until space or '>'
450  int nameEnd = nameStart;
451  while (nameEnd < soap.length()) {
452  char c = soap[nameEnd];
453  if (c == '>' || c == ' ' || c == '\t' || c == '\r' || c == '\n') break;
454  nameEnd++;
455  }
456  if (nameEnd <= nameStart) break;
457 
458  // extract name and value
459  Str nameStr;
460  nameStr.copyFrom(soap.c_str() + nameStart, nameEnd - nameStart);
461 
462  int valueStart = soap.indexOf('>', lt);
463  if (valueStart < 0) break;
464  valueStart += 1;
465  // closing tag for this element
466  char closeTag[220] = {0};
467  // use name without any namespace prefix
468  const char* fullName = nameStr.c_str();
469  const char* sep = strchr(fullName, ':');
470  const char* nameOnly = fullName;
471  if (sep) nameOnly = sep + 1;
472  snprintf(closeTag, sizeof(closeTag), "</%s>", nameOnly);
473  int closePos = soap.indexOf(closeTag, valueStart);
474  if (closePos < 0 || closePos > payloadEnd) break;
475 
476  Str valueStr;
477  valueStr.copyFrom(soap.c_str() + valueStart, closePos - valueStart);
478 
479  // trim whitespace
480  valueStr.trim();
481 
482  // store in string registry so pointers remain valid
483  const char* namePtr = strings.add((char*)nameOnly);
484  const char* valuePtr = strings.add((char*)valueStr.c_str());
485 
486  Argument arg;
487  arg.name = namePtr;
488  arg.value = valuePtr;
489  result.arguments.push_back(arg);
490 
491  pos = closePos + strlen(closeTag);
492  }
493  }
494  }
495  }
