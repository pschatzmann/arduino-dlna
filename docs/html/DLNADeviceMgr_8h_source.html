253  svcKey, cbStr.c_str(), tsec);
254  server->replyHeader().setValues(200, "OK");
255  server->replyHeader().put("SID", sid.c_str());
256  server->replyHeader().put("TIMEOUT", "Second-1800");
257  server->replyHeader().write(server->client());

262  return;
263  }

265  if (req.method() == T_POST) {
266  // Some stacks use POST for unsubscribe; try to handle SID
267  const char* sid = req.get("SID");

269  bool ok = DLNADeviceMgr::instance->subscriptionMgr.unsubscribe(

273  return;
274  }
275  }
276  }
277  // default reply OK
278  server->replyOK();
