185  // helper to write text content with minimal XML-escaping for &, < and >
186  static void writeEscapedText(Print& out, const char* s) {
187  if (!s) return;
188  for (const char* p = s; *p; ++p) {
189  if (*p == '&')
190  out.print("&amp;");
191  else if (*p == '<')
192  out.print("&lt;");
193  else if (*p == '>')
194  out.print("&gt;");
195  else
196  out.write((const uint8_t)*p);
197  }
198  }

200  // Print wrapper that escapes & < > while forwarding to an underlying Print
201  struct EscapingPrint : public Print {
202  Print& dest;
203  EscapingPrint(Print& d) : dest(d) {}
204  size_t write(uint8_t c) override {
205  if (c == '&') return dest.print("&amp;");
206  if (c == '<') return dest.print("&lt;");
207  if (c == '>') return dest.print("&gt;");
208  return dest.write(&c, 1);
209  }

211  size_t r = 0;
212  for (size_t i = 0; i < size; ++i) r += write(buffer[i]);

214  }

218  // streaming reply callback: writes the full SOAP envelope and escapes

220  static void streamReplyCallback(Print& out) {
221  XMLPrinter xml(out);
222  // Envelope start with namespace and encoding attribute
223  const char* envAttrs =
224  "xmlns:s=\"http://schemas.xmlsoap.org/soap/envelope/\" "
225  "s:encodingStyle=\"http://schemas.xmlsoap.org/soap/encoding/\"";
226  xml.printNodeBegin("Envelope", envAttrs, "s");
227  xml.printNodeBeginNl("Body", nullptr, "s");
228 
229  // BrowseResponse with its namespace declaration
230  const char* brAttrs =
231  "xmlns:u=\"urn:schemas-upnp-org:service:ContentDirectory:1\"";
232  xml.printNodeBeginNl("BrowseResponse", brAttrs, "u");
233 
234  // Result node: use callback to stream escaped DIDL
235  xml.printNode("Result", [&]() -> size_t {
236  if (g_stream_items)
237  buildEscapedDIDL(out, *g_stream_items);
238  else {
239  out.print(

241  "xmlns:upnp=\"urn:schemas-upnp-org:metadata-1-0/upnp/\" "
242  "xmlns=\"urn:schemas-upnp-org:metadata-1-0/DIDL-Lite/\"&gt;&lt;/"
243  "DIDL-Lite&gt;");
244  }
245  return 0;
246  });

251  xml.printNode("UpdateID", g_stream_updateID);

258  // helper: stream DIDL-Lite (escaped) for the provided items to the Print
259  static void buildEscapedDIDL(Print& out, Vector<MediaItem>& items) {
260  // Use XMLPrinter to generate DIDL elements, but wrap the output with
261  // EscapingPrint so '<' and '>' are emitted as escaped text into the
262  // surrounding SOAP <Result> element.
263  EscapingPrint esc(out);
264  XMLPrinter didl(esc);
265 
266  const char* didlAttrs =
267  "xmlns:dc=\"http://purl.org/dc/elements/1.1/\" "
268  "xmlns:upnp=\"urn:schemas-upnp-org:metadata-1-0/upnp/\" "
269  "xmlns=\"urn:schemas-upnp-org:metadata-1-0/DIDL-Lite/\"";
270  didl.printNodeBeginNl("DIDL-Lite", didlAttrs);
271 
272  for (auto it = items.begin(); it != items.end(); ++it) {
273  const MediaItem& item = *it;
274  char itemAttr[256];
275  snprintf(itemAttr, sizeof(itemAttr),
276  "id=\"%s\" parentID=\"%s\" restricted=\"%d\"",
277  item.id ? item.id : "", item.parentID ? item.parentID : "0",
278  item.restricted ? 1 : 0);
279  didl.printNodeBeginNl("item", itemAttr);
280 
281  // title
282  didl.printNode("dc:title", item.title ? item.title : "");
283 
284  // res with optional protocolInfo attribute
285  if (item.mimeType) {
286  char resAttr[128];
287  snprintf(resAttr, sizeof(resAttr), "protocolInfo=\"%s\"",
288  item.mimeType);
289  didl.printNode("res", item.res ? item.res : "", resAttr);
290  } else {
291  didl.printNode("res", item.res ? item.res : "");
292  }
293 
294  didl.printNodeEnd("item");
295  }
296 
297  didl.printNodeEnd("DIDL-Lite");
298  }
