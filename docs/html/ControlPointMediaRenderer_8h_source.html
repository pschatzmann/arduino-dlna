296  // Helper: parse DLNA time string (HH:MM:SS, MM:SS) to milliseconds
297  static unsigned long parseTimeToMs(const char* t) {
298  if (t == nullptr) return 0;
299  int h = 0, m = 0, s = 0;
300  int parts = 0;
301  // Count colons
302  const char* p = t;
303  int colonCount = 0;
304  while (*p) { if (*p == ':') colonCount++; p++; }
305  if (colonCount == 2) {
306  parts = sscanf(t, "%d:%d:%d", &h, &m, &s);
307  } else if (colonCount == 1) {
308  parts = sscanf(t, "%d:%d", &m, &s);
309  h = 0;
310  } else {
311  // maybe it's seconds only
312  parts = sscanf(t, "%d", &s);
313  }
314  if (parts <= 0) return 0;
315  unsigned long total = (unsigned long)h * 3600UL + (unsigned long)m * 60UL + (unsigned long)s;
316  return total * 1000UL;
317  }
