381  // Helper: parse DLNA time string (HH:MM:SS, MM:SS) to milliseconds
382  static unsigned long parseTimeToMs(const char* t) {
383  if (t == nullptr) return 0;
384  int h = 0, m = 0, s = 0;
385  int parts = 0;
386  // Count colons
387  const char* p = t;
388  int colonCount = 0;
389  while (*p) {
390  if (*p == ':') colonCount++;
391  p++;
392  }
393  if (colonCount == 2) {
394  parts = sscanf(t, "%d:%d:%d", &h, &m, &s);
395  } else if (colonCount == 1) {
396  parts = sscanf(t, "%d:%d", &m, &s);
397  h = 0;
398  } else {
399  // maybe it's seconds only
400  parts = sscanf(t, "%d", &s);
401  }
402  if (parts <= 0) return 0;
403  unsigned long total =
404  (unsigned long)h * 3600UL + (unsigned long)m * 60UL + (unsigned long)s;
405  return total * 1000UL;
406  }
