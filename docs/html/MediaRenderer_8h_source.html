255  static void transportControlCB(HttpServer* server, const char* requestPath,
256  HttpRequestHandlerLine* hl) {
257  // Parse SOAP request and extract action
258  Str soap = server->contentStr();
259  DlnaLogger.log(DlnaLogLevel::Info, "Transport Control: %s", soap.c_str());
260  MediaRenderer& media_renderer = *((MediaRenderer*)hl->context[0]);
261 
262  Str reply_str{reply()};
263  reply_str.replaceAll("%2", "AVTransport");
264  if (soap.indexOf("Play") >= 0) {
265  media_renderer.setActive(true);
266  reply_str.replaceAll("%1", "PlayResponse");
267  server->reply("text/xml", reply_str.c_str());
268  } else if (soap.indexOf("Pause") >= 0) {
269  media_renderer.setActive(false);
270  reply_str.replaceAll("%1", "PauseResponse");
271  server->reply("text/xml", reply_str.c_str());
272  } else if (soap.indexOf("Stop") >= 0) {
273  media_renderer.setActive(false);
274  reply_str.replaceAll("%1", "StopResponse");
275  server->reply("text/xml", reply_str.c_str());
276  } else if (soap.indexOf("SetAVTransportURI") >= 0) {
277  // Extract URL from SOAP request safely
278  int urlTagStart = soap.indexOf("<CurrentURI>");
279  int urlTagEnd = soap.indexOf("</CurrentURI>");
280  if (urlTagStart >= 0 && urlTagEnd > urlTagStart) {
281  int urlStart = urlTagStart + (int)strlen("<CurrentURI>");
282  Str url = soap.substring(urlStart, urlTagEnd);
283  media_renderer.play(url.c_str());
284  } else {
285  DlnaLogger.log(DlnaLogLevel::Warning,
286  "SetAVTransportURI called with invalid SOAP payload");
287  }
288  reply_str.replaceAll("%1", "SetAVTransportURIResponse");
289  server->reply("text/xml", reply_str.c_str());
290  } else {
291  // Handle other transport controls
292  reply_str.replaceAll("%1", "ActionResponse");
293  server->reply("text/xml", reply_str.c_str());
294  }
295  }
