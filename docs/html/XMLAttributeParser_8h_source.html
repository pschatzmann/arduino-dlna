8 // Small utility to extract attributes from a start-tag in an XML fragment.
9 // It does not attempt full XML validation but is safe for small embedded
10 // parsing tasks (e.g. DIDL <res ... protocolInfo="...">).
11 class XMLAttributeParser {
12  public:
13  // Find the first occurrence of `tagName` (e.g. "<res") in `xml` and
14  // extract the value of attribute `attrName` into `outBuf` (size `bufSize`).
15  // Returns true on success and fills outBuf (null-terminated). If the
16  // attribute value is longer than bufSize-1, it will be truncated.
17  static bool extractAttribute(const char* xml, const char* tagName,
18  const char* attrName, char* outBuf,
19  size_t bufSize) {
20  if (!xml || !tagName || !attrName || !outBuf || bufSize == 0) return false;
21  const char* p = strstr(xml, tagName);
22  while (p) {
23  const char* tagEnd = strchr(p, '>');
24  if (!tagEnd) return false;
25  const char* attrPos = strstr(p, attrName);
26  if (attrPos && attrPos < tagEnd) {
27  const char* q = strchr(attrPos, '"');
28  if (!q || q > tagEnd) return false;
29  q++;
30  const char* qend = strchr(q, '"');
31  if (!qend || qend > tagEnd) return false;
32  size_t vlen = (size_t)(qend - q);
33  size_t copyLen = vlen < (bufSize - 1) ? vlen : (bufSize - 1);
34  if (copyLen > 0) memcpy(outBuf, q, copyLen);
35  outBuf[copyLen] = '\0';
36  return true;
37  }
38  p = strstr(p + 1, tagName);
39  }
40  return false;
41  }

43  // Generic helper: extract the nth (1-based) colon-separated token from the
44  // value of attribute `attrName` on the first occurrence of `tagName`.
45  //
46  // Behavior and parameters:
47  // - `xml` : pointer to a null-terminated XML fragment to search.
48  // - `tagName` : start-tag to locate (for example "<res"). The first
49  // occurrence is scanned; the search does not validate full
50  // XML correctness.
51  // - `attrName` : attribute name to extract. Can be passed either with or
52  // without the trailing '=' (e.g. "protocolInfo" or
53  // "protocolInfo="). The function looks for the attribute
54  // inside the located start-tag and expects a double-quoted
55  // value: protocolInfo="...".
56  // - `n` : 1-based index of the colon-separated token to return.
57  // n==1 returns the first token. Tokens are split on ':'
58  // characters and are not trimmed; embedded XML entities are
59  // not decoded.
60  // - `outBuf`/`bufSize`: destination buffer and its size (must be > 0).
61  //
62  // Return value and truncation:
63  // - On success the requested token (non-empty) is copied into `outBuf`,
64  // null-terminated. If the token is longer than `bufSize - 1` it will be
65  // truncated to fit and still null-terminated. The function returns true
66  // when a non-empty token was extracted into `outBuf`.
67  // - The function returns false if any input is invalid, the tag or
68  // attribute cannot be found inside the tag, or the requested token index
69  // does not exist (or is empty).
70  //
71  // Example:
72  // attrName = "protocolInfo" (or "protocolInfo="), n = 3
73  // protocolInfo="http-get:*:audio/mpeg:*" -> returns "audio/mpeg".
74  static bool extractAttributeToken(const char* xml, const char* tagName,
75  const char* attrName, int n,
76  char* outBuf, size_t bufSize) {
77  if (!xml || !tagName || !attrName || n <= 0 || !outBuf || bufSize == 0) return false;
78  char proto[256] = {0};
79  bool ok = extractAttribute(xml, tagName, attrName, proto, sizeof(proto));
80  if (!ok || proto[0] == '\0') return false;
81  // proto format: token1:token2:token3:token4
82  int col = 1;
83  const char* tokenStart = proto;
84  if (n > 1) {
85  for (size_t i = 0; proto[i] != '\0'; ++i) {
86  if (proto[i] == ':') {
87  ++col;
88  if (col == n) {
89  tokenStart = proto + i + 1;
90  break;
91  }
92  }
93  }
94  }
95  if (!tokenStart || *tokenStart == '\0') return false;
96  // copy token into outBuf until ':' or end
97  size_t j = 0;
98  for (; tokenStart[j] != '\0' && tokenStart[j] != ':' && j + 1 < bufSize; ++j) outBuf[j] = tokenStart[j];
99  outBuf[j] = '\0';
100  return outBuf[0] != '\0';
101  }
102 };
