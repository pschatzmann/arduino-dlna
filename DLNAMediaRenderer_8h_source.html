265 // Transport control handler
266 static void transportControlCB(HttpServer* server, const char* requestPath,
267 HttpRequestHandlerLine* hl) {
268 // Incremental SOAP parsing using XMLParserPrint to avoid buffering the
269 // entire body in RAM. We detect the invoked action name and extract
270 // parameters like CurrentURI.
271 DLNAMediaRenderer& media_renderer = *((DLNAMediaRenderer*)hl->context[0]);
272
273 XMLParserPrint xp;
274 xp.setExpandEncoded(true);
275 Str nodeName;
276 Vector<Str> path{6};
277 Str text;
278 Str attrs;
279
280 Str actionName;
281 Str currentUri;
282
283 Client& client = server->client();
284 uint8_t buf[XML_PARSER_BUFFER_SIZE];
285 while (client.available()) {
286 int r = client.read(buf, XML_PARSER_BUFFER_SIZE);
287 if (r <= 0) break;
288 xp.write(buf, r);
289 while (xp.parse(nodeName, path, text, attrs)) {
290 // capture action name (Play, Pause, Stop, SetAVTransportURI etc.)
291 if (actionName.isEmpty()) {
292 if (strcmp(nodeName.c_str(), "Play") == 0 ||
293 strcmp(nodeName.c_str(), "Pause") == 0 ||
294 strcmp(nodeName.c_str(), "Stop") == 0 ||
295 strcmp(nodeName.c_str(), "SetAVTransportURI") == 0) {
296 actionName = nodeName;
297 }
298 }
299
300 // capture CurrentURI value
301 if (strcmp(nodeName.c_str(), "CurrentURI") == 0 && !text.isEmpty()) {
302 currentUri = text;
303 }
304 }
305 }
306
307 Str reply_str{reply()};
308 reply_str.replaceAll("%2", "AVTransport");
309
310 if (actionName == "Play") {
311 media_renderer.setActive(true);
312 reply_str.replaceAll("%1", "PlayResponse");
313 server->reply("text/xml", reply_str.c_str());
314 } else if (actionName == "Pause") {
315 media_renderer.setActive(false);
316 reply_str.replaceAll("%1", "PauseResponse");
317 server->reply("text/xml", reply_str.c_str());
318 } else if (actionName == "Stop") {
319 media_renderer.setActive(false);
320 reply_str.replaceAll("%1", "StopResponse");
321 server->reply("text/xml", reply_str.c_str());
322 } else if (actionName == "SetAVTransportURI") {
323 if (!currentUri.isEmpty()) {
324 media_renderer.play(currentUri.c_str());
325 } else {
326 DlnaLogger.log(DlnaLogLevel::Warning,
327 "SetAVTransportURI called with invalid SOAP payload");
328 }
329 reply_str.replaceAll("%1", "SetAVTransportURIResponse");
330 server->reply("text/xml", reply_str.c_str());
331 } else {
332 // Handle other transport controls
333 reply_str.replaceAll("%1", "ActionResponse");
334 server->reply("text/xml", reply_str.c_str());
335 }
336 }

338 // Rendering control handler
339 static void renderingControlCB(HttpServer* server, const char* requestPath,
340 HttpRequestHandlerLine* hl) {
341 // Incremental SOAP parsing: detect SetVolume/SetMute and extract values
342 DLNAMediaRenderer& media_renderer = *((DLNAMediaRenderer*)hl->context[0]);
343
344 XMLParserPrint xp;
345 xp.setExpandEncoded(true);
346 Str nodeName;
347 Vector<Str> path{6};
348 Str text;
349 Str attrs;
350
351 Str actionName;
352 int desiredVolume = -1;
353 bool desiredMute = false;
354 bool haveVolume = false;
355 bool haveMute = false;
356
357 Client& client = server->client();
358 uint8_t buf[XML_PARSER_BUFFER_SIZE];
359 while (client.available()) {
360 int r = client.read(buf, XML_PARSER_BUFFER_SIZE);
361 if (r <= 0) break;
362 xp.write(buf, r);
363 while (xp.parse(nodeName, path, text, attrs)) {
364 if (actionName.isEmpty()) {
365 if (strcmp(nodeName.c_str(), "SetVolume") == 0 ||
366 strcmp(nodeName.c_str(), "SetMute") == 0) {
367 actionName = nodeName;
368 }
369 }
370 if (strcmp(nodeName.c_str(), "DesiredVolume") == 0 && !text.isEmpty()) {
371 desiredVolume = text.toInt();
372 haveVolume = true;
373 }
374 if (strcmp(nodeName.c_str(), "DesiredMute") == 0 && !text.isEmpty()) {
375 desiredMute = (text == "1");
376 haveMute = true;
377 }
378 }
379 }
380
381 Str reply_str{reply()};
382 reply_str.replaceAll("%2", "RenderingControl");
383
384 if (actionName == "SetVolume" && haveVolume) {
385 media_renderer.setVolume((uint8_t)desiredVolume);
386 reply_str.replaceAll("%1", "SetVolumeResponse");
387 server->reply("text/xml", reply_str.c_str());
388 } else if (actionName == "SetMute" && haveMute) {
389 media_renderer.setMute(desiredMute);
390 reply_str.replaceAll("%1", "SetMuteResponse");
391 server->reply("text/xml", reply_str.c_str());
392 } else {
393 reply_str.replaceAll("%1", "RenderingControl");
394 server->reply("text/xml", reply_str.c_str());
395 }
