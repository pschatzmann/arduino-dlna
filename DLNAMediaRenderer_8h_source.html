451 static int printReplyXML(Print& out, const char* replyName,
452 const char* serviceId,
453 const char* values = nullptr) {
454 XMLPrinter xp(out);
455 size_t result = 0;
456 result += xp.printNodeBegin(
457 "s:Envelope", "xmlns:s=\"http://schemas.xmlsoap.org/soap/envelope/\"");
458 result += xp.printNodeBegin("s:Body");
459 result += xp.printf("<u:%s xmlns:u=\"urn:schemas-upnp-org:service:%s:1\">",
460 replyName, serviceId);
461
462 // e.g.<u:return>Stop,Pause,Next,Seek</u:return> for
463 // QueryStateVariableResponse
464 if (values) {
465 result += xp.printf("%s", values);
466 }
467
468 result += xp.printf("</u:%s>", replyName);
469 result += xp.printNodeEnd("s:Body");
470 result += xp.printNodeEnd("s:Envelope");
471 return result;
472 }

504 bool processAction(ActionRequest& action, HttpServer& server) {
505 DlnaLogger.log(DlnaLogLevel::Info, "DLNAMediaRenderer::processAction: %s",
506 action.action);
507 StrView action_str(action.action);
508 if (action_str.isEmpty()) {
509 DlnaLogger.log(DlnaLogLevel::Error, "Empty action received");
510 server.replyError(400, "Invalid Action");
511 return false;
512 }
513
514 for (const auto& rule : rules) {
515 if (action_str.endsWith(rule.suffix)) {
516 return rule.handler(this, action, server);
517 }
518 }
519 DlnaLogger.log(DlnaLogLevel::Error, "Unsupported action: %s",
520 action.action);
521 server.replyError(400, "Invalid Action");
522 return false;
523 }
