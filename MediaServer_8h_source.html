276 bool processAction(ActionRequest& action, HttpServer& server) {
277 DlnaLogger.log(DlnaLogLevel::Info, "processAction");
278 StrView action_str(action.action);
279 if (action_str.isEmpty()) {
280 DlnaLogger.log(DlnaLogLevel::Error, "Empty action received");
281 server.replyNotFound();
282 return false;
283 }
284 if (action_str.endsWith("Browse")) {
285 return processActionBrowse(action, server);
286 } else if (action_str.endsWith("Search")) {
287 return processActionSearch(action, server);
288 } else if (action_str.endsWith("GetSearchCapabilities")) {
289 return processActionGetSearchCapabilities(action, server);
290 } else if (action_str.endsWith("GetSortCapabilities")) {
291 return processActionGetSortCapabilities(action, server);
292 } else if (action_str.endsWith("GetSystemUpdateID")) {
293 return processActionGetSystemUpdateID(action, server);
294 } else {
295 DlnaLogger.log(DlnaLogLevel::Error, "Unsupported action: %s",
296 action.action);
297 return false;
298 server.replyNotFound();
299 }
300 }

538 bool streamActionItems(const char* responseName, const char* objectID,
539 const char* browseFlag, const char* filter,
540 int startingIndex, int requestedCount,
541 HttpServer& server) {
542 int numberReturned = 0;
543 int totalMatches = 0;
544 int updateID = g_stream_updateID;
545
546 // allow caller to prepare counts/updateID
547 if (prepare_data_cb) {
548 prepare_data_cb(objectID, browseFlag, filter, startingIndex,
549 requestedCount, nullptr, numberReturned, totalMatches,
550 updateID, reference_);
551 }
552
553 ChunkPrint chunk_writer{server.client()};
554 server.replyChunked("text/xml");
555
556 // SOAP envelope start and response wrapper
557 soapEnvelopeStart(chunk_writer);
558 // responseName is assumed to be e.g. "BrowseResponse" or "SearchResponse"
559 actionResponseStart(chunk_writer, responseName,
560 "urn:schemas-upnp-org:service:ContentDirectory:1");
561
562 // Result -> DIDL-Lite
563 chunk_writer.println("<Result>");
564 // DIDL root with namespaces
565 streamDIDL(chunk_writer, numberReturned, startingIndex);
566 chunk_writer.print("</Result>\r\n");
567
568 // numeric fields

581 void streamDIDL(ChunkPrint& chunk_writer, int numberReturned,
582 int startingIndex) {
583 // Stream DIDL-Lite Result payload using helper
584 chunk_writer.print(
585 "<DIDL-Lite xmlns:dc=\"http://purl.org/dc/elements/1.1/\" "
586 "xmlns:upnp=\"urn:schemas-upnp-org:metadata-1-0/upnp/\" "
587 "xmlns=\"urn:schemas-upnp-org:metadata-1-0/DIDL-Lite/\">\r\n");
588
589 if (get_data_cb) {
590 for (int i = 0; i < numberReturned; ++i) {
591 MediaItem item;
592 int idx = startingIndex + i;
593 if (!get_data_cb(idx, item, reference_)) break;
594
595 chunk_writer.printf(
596 "<item id=\"%s\" parentID=\"%s\" restricted=\"%d\">",
597 item.id ? item.id : "", item.parentID ? item.parentID : "0",
598 item.restricted ? 1 : 0);
599
600 // title
601 chunk_writer.print("<dc:title>");
602 chunk_writer.print(nullStr(item.title));
603 chunk_writer.print("</dc:title>\r\n");
604
605 // res with optional protocolInfo attribute
606 if (item.mimeType) {
607 chunk_writer.printf("<res protocolInfo=\"%s\">", item.mimeType);
608 chunk_writer.print(nullStr(item.res));
609 chunk_writer.print("</res>\r\n");
610 } else {
611 chunk_writer.print("<res>");
612 chunk_writer.print(nullStr(item.res));
613 chunk_writer.print("</res>\r\n");
614 }
615
616 chunk_writer.print("</item>\r\n");
617 }
618 }
619
620 chunk_writer.print("</DIDL-Lite>\r\n");
621 }

646 HttpRequestHandlerLine* hl) {
647 // Parse the incoming SOAP action
648 ActionRequest action;
649 DLNADevice::parseActionRequest(server, requestPath, hl, action);
650
651 MediaServer* ms = nullptr;
652 if (hl && hl->context[0]) ms = (MediaServer*)hl->context[0];
653
654 if (!ms) {
655 server->replyNotFound();
656 return;
657 }
658
659 // Dispatch to instance methods for clarity and to allow overrides
660 if (StrView(action.action).endsWith("GetProtocolInfo")) {
661 ms->processActionGetProtocolInfo(action, *server);
662 return;
663 } else if (StrView(action.action).endsWith("GetCurrentConnectionIDs")) {
664 ms->processActionGetCurrentConnectionIDs(action, *server);
665 return;
666 } else if (StrView(action.action).endsWith("GetCurrentConnectionInfo")) {
667 ms->processActionGetCurrentConnectionInfo(action, *server);
668 return;
669 }
670
671 // Unhandled actions: reply NotFound
672 server->replyNotFound();
673 }
