133 bool loop() override {
134 if (!is_active) return false;
135 // Platform-specific periodic diagnostics (e.g. ESP32 memory logging)
136 logScheduledStatus();
137
138 // handle server requests
139 bool rc = loopServer();
140
141 // Use millisecond-based intervals for scheduling so callers can set
142 // real-time intervals instead of loop-counts.
143 uint64_t now = millis();
144 if (isSchedulerActive() && now >= next_scheduler_timeout_ms) {
145 processUdpAndScheduler();
146 // schedule next run
147 next_scheduler_timeout_ms = now + scheduler_interval_ms;
148 }
149
150 // deliver any queued subscription notifications (if enabled)
151 if (isSubscriptionsActive() && now >= next_subscriptions_timeout_ms) {
152 subscription_mgr.publish();
153 // schedule next run
154 next_subscriptions_timeout_ms = now + subscriptions_interval_ms;
155 }
156
157 // be nice, if we have other tasks
158 delay(DLNA_LOOP_DELAY_MS);
159 return true;
160 }

235 static void parseActionRequest(IHttpServer* server, const char* requestPath,
236 HttpRequestHandlerLine* hl,
237 ActionRequest& action) {
238 DlnaLogger.log(DlnaLogLevel::Info, "parseActionRequest");
239 auto start = millis();
240 XMLParserPrint xp;
241 xp.setExpandEncoded(true);
242
243 Str outNodeName;
244 Vector<Str> outPath;
245 Str outText;
246 Str outAttributes;
247 bool is_attribute = false;
248 bool is_action = false;
249 Str actionName;
250 char buffer[XML_PARSER_BUFFER_SIZE];
251 Client& client = server->client();
252
253 client.setTimeout(20);
254 while (true) {
255 size_t len = client.readBytes(buffer, sizeof(buffer));
256 if (len == 0) break;
257 xp.write((const uint8_t*)buffer, len);
258
259 while (xp.parse(outNodeName, outPath, outText, outAttributes)) {
260 if (is_attribute) {
261 const char* argName = outNodeName.c_str();
262 action.addArgument(argName, outText.c_str());
263 continue;
264 }
265 if (is_action) {
266 is_action = false;
267 is_attribute = true;
268 action.setAction(outNodeName.c_str());
269 DlnaLogger.log(DlnaLogLevel::Info, "action: %s", action.getAction());
270 continue;
271 }
272 // skip SOAP envelope wrappers
273 if (outNodeName.equals("s:Envelope") || outNodeName.equals("Body")) {
274 continue;
275 }
276 if (outNodeName.equals("s:Body")) {
277 is_action = true;
278 }
279 }
280 }
281 xp.end();
282 DlnaLogger.log(DlnaLogLevel::Info, "Parse took %d ms", millis() - start);
283 }

293 static size_t printReplyXML(
294 Print& out, const char* replyName, const char* serviceId,
295 std::function<size_t(Print&, void*)> valuesWriter = nullptr,
296 void* ref = nullptr) {
297 XMLPrinter xp(out);
298 size_t result = 0;
299 result += xp.printNodeBegin(
300 "s:Envelope", "xmlns:s=\"http://schemas.xmlsoap.org/soap/envelope/\"");
301 result += xp.printNodeBegin("s:Body");
302 result += xp.printf("<u:%s xmlns:u=\"urn:schemas-upnp-org:service:%s:1\">",
303 replyName, serviceId);
304
305 // e.g.<u:return>Stop,Pause,Next,Seek</u:return> for
306 // QueryStateVariableResponse
307 if (valuesWriter) {
308 result += valuesWriter(out, ref);
309 }
310
311 result += xp.printf("</u:%s>", replyName);
312 result += xp.printNodeEnd("s:Body");
313 result += xp.printNodeEnd("s:Envelope");
314 return result;
315 }

380 static bool handleSubscription(IHttpServer* server, const char* requestPath,
381 HttpRequestHandlerLine* hl,
382 bool& is_subscribe) {
383 // Dispatch to dedicated handlers for subscribe/unsubscribe to keep the
384 // logic separated and easier to test.
385 is_subscribe = false;
386 HttpRequestHeader& req = server->requestHeader();
387 if (req.method() == T_SUBSCRIBE) {
388 is_subscribe = true;
389 return handleSubscribe(server, requestPath, hl);
390 }
391 if (req.method() == T_UNSUBSCRIBE) {
392 return handleUnsubscribe(server, requestPath, hl);
393 }
394
395 // default reply OK for other methods
396 server->replyError(501, "Unsupported Method");
397 return false;
398 }

518 bool validateServiceInfo(const DLNAServiceInfo& service) {
519 bool has_error = false;
520
521 if (!service.scpd_url || !*service.scpd_url) {
522 DlnaLogger.log(DlnaLogLevel::Error, "Service missing scpd_url");
523 has_error = true;
524 }
525 if (!service.control_url || !*service.control_url) {
526 DlnaLogger.log(DlnaLogLevel::Error, "Service missing control_url");
527 has_error = true;
528 }
529 if (!service.event_sub_url || !*service.event_sub_url) {
530 DlnaLogger.log(DlnaLogLevel::Error, "Service missing event_sub_url");
531 has_error = true;
532 }
533 if (!service.scp_cb) {
534 DlnaLogger.log(DlnaLogLevel::Error, "Service missing scp_cb for %s",
535 service.scpd_url ? service.scpd_url : "(null)");
536 has_error = true;
537 }
538 if (!service.control_cb) {
539 DlnaLogger.log(DlnaLogLevel::Error, "Service missing control_cb for %s",
540 service.control_url ? service.control_url : "(null)");
541 has_error = true;
542 }
543 if (!service.event_sub_cb) {
544 DlnaLogger.log(DlnaLogLevel::Error, "Service missing event_sub_cb for %s",
545 service.event_sub_url ? service.event_sub_url : "(null)");
546 has_error = true;
547 }
548
549 if (has_error) {
550 DlnaLogger.log(DlnaLogLevel::Error,
551 "Service validation failed - missing URLs/callbacks");
552 }
553
554 return !has_error;
555 }
