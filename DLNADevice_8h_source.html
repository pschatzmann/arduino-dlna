152 static void eventSubscriptionHandler(HttpServer* server,
153 const char* requestPath,
154 HttpRequestHandlerLine* hl) {
155 DlnaLogger.log(DlnaLogLevel::Debug, "eventSubscriptionHandler");
156 // context[0] contains DLNADeviceInfo*
157 DLNADeviceInfo* device = (DLNADeviceInfo*)(hl->context[0]);
158 if (!device) {
159 server->replyNotFound();
160 return;
161 }
162 // header parsing
163 HttpRequestHeader& req = server->requestHeader();
164 // SUBSCRIBE handling
165 if (req.method() == T_SUBSCRIBE) {
166 const char* cb = req.get("CALLBACK");
167 const char* timeout = req.get("TIMEOUT");
168 Str cbStr;
169 if (cb) {
170 cbStr = cb;
171 cbStr.replace("<", "");
172 cbStr.replace(">", "");
173 }
174 uint32_t tsec = 1800;
175 if (timeout && StrView(timeout).startsWith("Second-")) {
176 tsec = atoi(timeout + 7);
177 }
178 // Use the service event path as key for subscriptions
179 const char* svcKey = requestPath;
180 if (DLNADevice::instance) {
181 Str sid = DLNADevice::instance->subscriptionMgr.subscribe(
182 svcKey, cbStr.c_str(), tsec);
183 server->replyHeader().setValues(200, "OK");
184 server->replyHeader().put("SID", sid.c_str());
185 server->replyHeader().put("TIMEOUT", "Second-1800");
186 server->replyHeader().write(server->client());
187 server->endClient();
188 return;
189 }
190 server->replyNotFound();
191 return;
192 }
193 // UNSUBSCRIBE: handle via SID header
194 if (req.method() == T_POST) {
195 // Some stacks use POST for unsubscribe; try to handle SID
196 const char* sid = req.get("SID");
197 if (sid && DLNADevice::instance) {
198 bool ok =
199 DLNADevice::instance->subscriptionMgr.unsubscribe(requestPath, sid);
200 if (ok) {
201 server->replyOK();
202 return;
203 }
204 }
205 }
206 // default reply OK
207 server->replyOK();
208 }

211 static void parseActionRequest(HttpServer* server, const char* requestPath,
212 HttpRequestHandlerLine* hl,
213 ActionRequest& action) {
214 DlnaLogger.log(DlnaLogLevel::Info, "parseActionRequest");
215
216 XMLParserPrint xp;
217 xp.setExpandEncoded(true);
218
219 Str outNodeName;
220 Vector<Str> outPath;
221 Str outText;
222 Str outAttributes;
223 bool is_attribute = false;
224 bool is_action = false;
225 Str actionName;
226 char buffer[256];
227 Client &client = server->client();
228
229 while (client.available() > 0) {
230 size_t len = client.readBytes(buffer, sizeof(buffer));
231 xp.write((const uint8_t*)buffer, len);
232
233 while (xp.parse(outNodeName, outPath, outText, outAttributes)) {
234 if (is_attribute) {
235 const char* argName = registry.add((char*)outNodeName.c_str());
236 action.addArgument(argName, outText.c_str());
237
238 continue;
239 }
240 if (is_action) {
241 is_action = false;
242 is_attribute = true;
243 action.action = registry.add((char*)outNodeName.c_str());
244 DlnaLogger.log(DlnaLogLevel::Info, "action: %s", action.action);
245 continue;
246 }
247 // skip SOAP envelope wrappers
248 if (outNodeName.equals("s:Envelope") || outNodeName.equals("Body")) {
249 continue;
250 }
251 if (outNodeName.equals("s:Body")) {
252 is_action = true;
253 }
254 }
255 }
