121 bool loop() {
122 if (!is_active) return false;
123 // Platform-specific periodic diagnostics (e.g. ESP32 memory logging)
124 logMemoryIfNeeded();
125
126 // handle server requests
127 bool rc = loopServer();
128
129 // Use millisecond-based intervals for scheduling so callers can set
130 // real-time intervals instead of loop-counts.
131 uint64_t now = millis();
132 if (isSchedulerActive() && now >= next_scheduler_timeout_ms) {
133 processUdpAndScheduler();
134 // schedule next run
135 next_scheduler_timeout_ms = now + scheduler_interval_ms;
136 }
137
138 // deliver any queued subscription notifications (if enabled)
139 if (isSubscriptionsActive() && now >= next_subscriptions_timeout_ms) {
140 subscription_mgr.publish();
141 // schedule next run
142 next_subscriptions_timeout_ms = now + subscriptions_interval_ms;
143 }
144
145 // be nice, if we have other tasks
146 delay(DLNA_LOOP_DELAY_MS);
147 return true;
148 }

222 static void parseActionRequest(HttpServer* server, const char* requestPath,
223 HttpRequestHandlerLine* hl,
224 ActionRequest& action) {
225 DlnaLogger.log(DlnaLogLevel::Info, "parseActionRequest");
226 auto start = millis();
227 XMLParserPrint xp;
228 xp.setExpandEncoded(true);
229
230 Str outNodeName;
231 Vector<Str> outPath;
232 Str outText;
233 Str outAttributes;
234 bool is_attribute = false;
235 bool is_action = false;
236 Str actionName;
237 char buffer[XML_PARSER_BUFFER_SIZE];
238 Client& client = server->client();
239
240 client.setTimeout(20);
241 while (true) {
242 size_t len = client.readBytes(buffer, sizeof(buffer));
243 if (len == 0) break;
244 xp.write((const uint8_t*)buffer, len);
245
246 while (xp.parse(outNodeName, outPath, outText, outAttributes)) {
247 if (is_attribute) {
248 const char* argName = registry.add((char*)outNodeName.c_str());
249 action.addArgument(argName, outText.c_str());
250 continue;
251 }
252 if (is_action) {
253 is_action = false;
254 is_attribute = true;
255 action.setAction(outNodeName.c_str());
256 DlnaLogger.log(DlnaLogLevel::Info, "action: %s", action.getAction());
257 continue;
258 }
259 // skip SOAP envelope wrappers
260 if (outNodeName.equals("s:Envelope") || outNodeName.equals("Body")) {
261 continue;
262 }
263 if (outNodeName.equals("s:Body")) {
264 is_action = true;
265 }
266 }
267 }
268 xp.end();
269 DlnaLogger.log(DlnaLogLevel::Info, "Parse took %d ms", millis() - start);
270 }

280 static size_t printReplyXML(
281 Print& out, const char* replyName, const char* serviceId,
282 std::function<size_t(Print&, void*)> valuesWriter = nullptr,
283 void* ref = nullptr) {
284 XMLPrinter xp(out);
285 size_t result = 0;
286 result += xp.printNodeBegin(
287 "s:Envelope", "xmlns:s=\"http://schemas.xmlsoap.org/soap/envelope/\"");
288 result += xp.printNodeBegin("s:Body");
289 result += xp.printf("<u:%s xmlns:u=\"urn:schemas-upnp-org:service:%s:1\">",
290 replyName, serviceId);
291
292 // e.g.<u:return>Stop,Pause,Next,Seek</u:return> for
293 // QueryStateVariableResponse
294 if (valuesWriter) {
295 result += valuesWriter(out, ref);
296 }
297
298 result += xp.printf("</u:%s>", replyName);
299 result += xp.printNodeEnd("s:Body");
300 result += xp.printNodeEnd("s:Envelope");
301 return result;
302 }

367 static void eventSubscriptionHandler(HttpServer* server,
368 const char* requestPath,
369 HttpRequestHandlerLine* hl) {
370 // Dispatch to dedicated handlers for subscribe/unsubscribe to keep the
371 // logic separated and easier to test.
372 HttpRequestHeader& req = server->requestHeader();
373 if (req.method() == T_SUBSCRIBE) {
374 handleSubscribe(server, requestPath, hl);
375 // default reply OK for other methods
376 server->replyOK();
377 return;
378 }
379 if (req.method() == T_UNSUBSCRIBE) {
380 handleUnsubscribe(server, requestPath, hl);
381 // default reply OK for other methods
382 server->replyOK();
383 return;
384 }
385
386 // default reply OK for other methods
387 server->replyError(501, "Unsupported Method");
388 }

582 static void handleSubscribe(HttpServer* server, const char* requestPath,
583 HttpRequestHandlerLine* hl) {
584 DlnaLogger.log(DlnaLogLevel::Debug, "handleSubscribe");
585 DLNADevice* p_device = (DLNADevice*)server->getReference();
586 // context[0] contains DLNADeviceInfo*
587 DLNADeviceInfo* p_device_info = (DLNADeviceInfo*)(hl->context[0]);
588 if (!p_device_info) {
589 server->replyNotFound();
590 return;
591 }
592
593 const char* cb = server->requestHeader().get("CALLBACK");
594 const char* timeout = server->requestHeader().get("TIMEOUT");
595 Str req_sid = server->requestHeader().get("SID");
596 Str cbStr;
597 if (cb) {
598 cbStr = cb;
599 cbStr.replace("<", "");
600 cbStr.replace(">", "");
601 }
602 uint32_t tsec = 1800;
603 if (timeout && StrView(timeout).startsWith("Second-")) {
604 tsec = atoi(timeout + 7);
605 }
606
607 DLNAServiceInfo* svc = p_device->getServiceByEventPath(requestPath);
608 if (svc != nullptr) {
609 Str sid = p_device->subscription_mgr.subscribe(*svc, cbStr.c_str(),
610 req_sid.c_str(), tsec);
611 server->replyHeader().setValues(200, "OK");
612 server->replyHeader().put("SID", sid.c_str());
613 server->replyHeader().put("TIMEOUT", "Second-1800");
614 server->replyHeader().write(server->client());
615 server->replyOK();
616 return;
617 }
618
619 server->replyNotFound();
620 }

623 static void handleUnsubscribe(HttpServer* server, const char* requestPath,
624 HttpRequestHandlerLine* hl) {
625 DlnaLogger.log(DlnaLogLevel::Debug, "handleUnsubscribe");
626 DLNADevice* p_device = (DLNADevice*)server->getReference();
627
628 // context[0] contains DLNADeviceInfo*
629 DLNADeviceInfo* p_device_info = (DLNADeviceInfo*)(hl->context[0]);
630 if (!p_device_info) {
631 server->replyNotFound();
632 return;
633 }
634
635 const char* sid = server->requestHeader().get("SID");
636 if (sid && p_device) {
637 DLNAServiceInfo* svc = p_device->getServiceByEventPath(requestPath);
638 if (svc != nullptr) {
639 bool ok = p_device->subscription_mgr.unsubscribe(*svc, sid);
640 if (ok) {
641 server->replyOK();
642 return;
643 }
644 }
645 }
646 // fallthrough: reply OK by default
647 server->replyOK();
648 }
