189 bool onRequest(const char* path) override {
190 DlnaLogger.log(DlnaLogLevel::Info, "Serving at %s", path);
191
192 bool result = false;
193 // check in registered handlers
194 StrView pathStr = StrView(path);
195 pathStr.replace("//", "/"); // TODO investiage why we get //
196 for (auto handler_line_ptr : handler_collection) {
197 DlnaLogger.log(DlnaLogLevel::Debug, "onRequest: %s %s vs: %s %s %s", path,
198 methods[request_header.method()],
199 nullstr(handler_line_ptr->path.c_str()),
200 methods[handler_line_ptr->method],
201 nullstr(handler_line_ptr->mime));
202
203 if (pathStr.matches(handler_line_ptr->path.c_str()) &&
204 request_header.method() == handler_line_ptr->method &&
205 matchesMime(nullstr(handler_line_ptr->mime),
206 nullstr(request_header.accept()))) {
207 // call registed handler function
208 DlnaLogger.log(DlnaLogLevel::Debug, "onRequest %s", "->found",
209 nullstr(handler_line_ptr->path.c_str()));
210 handler_line_ptr->fn(this, path, handler_line_ptr);
211 result = true;
212 break;
213 }
214 }
215
216 if (!result) {
217 DlnaLogger.log(DlnaLogLevel::Error, "Request %s not available", path);
218 }
219
220 return result;

353 bool copy() override {
354 bool result = false;
355 if (!is_active) {
356 delay(no_connect_delay);
357 return false;
358 }
359
360 // Accept new client and add to list if connected
361 ClientType client = server_ptr->accept();
362 if (client.connected()) {
363 DlnaLogger.log(DlnaLogLevel::Info, "copy: accepted new client");
364 client.setTimeout(DLNA_HTTP_REQUEST_TIMEOUT_MS);
365 open_clients.push_back(client);
366 }
367
368 // Stop when nothing to process
369 if (open_clients.empty()) {
370 // delay(no_connect_delay);
371 return false;
372 }
373
374 if (current_client_iterator == open_clients.end()) {
375 current_client_iterator = open_clients.begin();
376 if (current_client_iterator == open_clients.end()) {
377 delay(no_connect_delay);
378 return false;
379 }
380 }
381
382 client_ptr = &(*current_client_iterator);
383 if (!client_ptr->connected()) {
384 DlnaLogger.log(DlnaLogLevel::Debug, "copy: removing disconnected client");
385 current_client_iterator = open_clients.erase(current_client_iterator);
386 // Do not advance iterator here; erase returns the next valid iterator
387 return false;
388 }
389
390 processRequest();
391 result = true;
392 ++current_client_iterator;
393 if (current_client_iterator == open_clients.end() &&
394 !open_clients.empty()) {
395 current_client_iterator = open_clients.begin();
396 }
397
398 // cleanup clients
399 removeClosedClients();
400
401 return result;
402 }

457 /* Remove all closed/disconnected clients from the open_clients list.
458 * This also keeps `current_client_iterator` valid by advancing it to
459 * the next element when the erased element equals the current iterator.
460 */
461 void removeClosedClients() {
462 auto it = open_clients.begin();
463 while (it != open_clients.end()) {
464 if ((!(*it).connected())) {
465 auto next = open_clients.erase(it);
466 if (current_client_iterator == it) {
467 current_client_iterator = next;
468 }
469 it = next;
470 } else {
471 ++it;
472 }
473 }
474 if (open_clients.empty()) {
475 current_client_iterator = open_clients.begin();
476 }

490 void processRequest() {
491 DlnaLogger.log(DlnaLogLevel::Info, "processRequest");
492 request_header.read(this->client());

506 const char* resolveRewrite(const char* from) {
507 for (auto i = rewrite_collection.begin(); i != rewrite_collection.end();
508 ++i) {
509 HttpRequestRewrite* rewrite = (*i);
510 if (rewrite->from.matches(from)) {
511 return rewrite->to.c_str();
512 }
513 }
