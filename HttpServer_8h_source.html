236 bool onRequest(const char* path) {
237 DlnaLogger.log(DlnaLogLevel::Info, "Serving at %s", path);
238
239 bool result = false;
240 // check in registered handlers
241 StrView pathStr = StrView(path);
242 pathStr.replace("//", "/"); // TODO investiage why we get //
243 for (auto handler_line_ptr : handler_collection) {
244 DlnaLogger.log(DlnaLogLevel::Debug, "onRequest: %s %s vs: %s %s %s", path,
245 methods[request_header.method()],
246 nullstr(handler_line_ptr->path.c_str()),
247 methods[handler_line_ptr->method],
248 nullstr(handler_line_ptr->mime));
249
250 if (pathStr.matches(handler_line_ptr->path.c_str()) &&
251 request_header.method() == handler_line_ptr->method &&
252 matchesMime(nullstr(handler_line_ptr->mime),
253 nullstr(request_header.accept()))) {
254 // call registed handler function
255 DlnaLogger.log(DlnaLogLevel::Debug, "onRequest %s", "->found",
256 nullstr(handler_line_ptr->path.c_str()));
257 handler_line_ptr->fn(this, path, handler_line_ptr);
258 result = true;
259 break;
260 }
261 }
262
263 if (!result) {
264 DlnaLogger.log(DlnaLogLevel::Error, "Request %s not available", path);
265 }

430 bool copy() {
431 bool result = false;
432 if (!is_active) {
433 delay(no_connect_delay);
434 return false;
435 }
436
437 // Accept new client and add to list if connected
438 WiFiClient client = server_ptr->accept();
439 if (client.connected()) {
440 DlnaLogger.log(DlnaLogLevel::Info, "copy: accepted new client");
441 open_clients.push_back(client);
442 }
443
444 // Stop when nothing to process
445 if (open_clients.empty()) {
446 delay(no_connect_delay);
447 return false;
448 }
449
450 if (current_client_iterator == open_clients.end()) {
451 current_client_iterator = open_clients.begin();
452 if (current_client_iterator == open_clients.end()) {
453 delay(no_connect_delay);
454 return false;
455 }
456 }
457
458 client_ptr = &(*current_client_iterator);
459 if (!client_ptr->connected()) {
460 DlnaLogger.log(DlnaLogLevel::Info, "copy: removing disconnected client");
461 current_client_iterator = open_clients.erase(current_client_iterator);
462 // Do not advance iterator here; erase returns the next valid iterator
463 return false;
464 }
465
466 processRequest();
467 result = true;
468 ++current_client_iterator;
469 if (current_client_iterator == open_clients.end() && !open_clients.empty()) {
470 current_client_iterator = open_clients.begin();
471 }
472
473 return result;
474 }
