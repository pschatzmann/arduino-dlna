124 bool browse(int startingIndex, int requestedCount, ContentQueryType queryType, XMLCallback XMLCallback,
125 int& numberReturned, int& totalMatches, int& updateID) {
126
127 // Register item callback
128 this->mgr.onResultNode(XMLCallback);
129 // Build and post browse action
130 DLNAServiceInfo& svc =
131 selectService("urn:upnp-org:serviceId:ContentDirectory");
132 if (!svc) return false;
133 ActionRequest& act =
134 createBrowseAction(svc, queryType, startingIndex, requestedCount);
135 mgr.addAction(act);
136 last_reply = mgr.executeActions();
137 if (!last_reply) return false;
138
139 // parse numeric result fields
140 parseNumericFields(last_reply, numberReturned, totalMatches, updateID);
141
142 return true;
143 }

303 ActionRequest &createBrowseAction(DLNAServiceInfo& svc, ContentQueryType queryType,
304 int startingIndex, int requestedCount) {
305 const char* browseFlag = (queryType == ContentQueryType::BrowseMetadata)
306 ? "BrowseMetadata"
307 : "BrowseDirectChildren";
308 static ActionRequest act(svc, "Browse");
309 // Use the canonical argument name expected by ContentDirectory: "ObjectID"
310 act.addArgument("ObjectID", object_id);
311 act.addArgument("BrowseFlag", browseFlag );
312 act.addArgument("Filter", "");
313 char buf[32];
314 snprintf(buf, sizeof(buf), "%d", startingIndex);
315 act.addArgument("StartingIndex", buf);
316 snprintf(buf, sizeof(buf), "%d", requestedCount);
317 act.addArgument("RequestedCount", buf);
318 act.addArgument("SortCriteria", "");
319 return act;
320 }
