153 bool browse(int startingIndex, int requestedCount, ContentQueryType queryType,
154 XMLCallback XMLCallback, int& numberReturned, int& totalMatches,
155 int& updateID) {
156 // Register item callback
157 this->p_mgr->onResultNode(XMLCallback);
158 // Build and post browse action
159 DLNAServiceInfo& svc =
160 selectService("urn:upnp-org:serviceId:ContentDirectory");
161 if (!svc) return false;
162 ActionRequest& act =
163 createBrowseAction(svc, queryType, startingIndex, requestedCount);
164 p_mgr->addAction(act);
165 last_reply = p_mgr->executeActions();
166 if (!last_reply) return false;
167
168 // parse numeric result fields
169 parseNumericFields(last_reply, numberReturned, totalMatches, updateID);
170
171 return true;
172 }

328 ActionRequest& createBrowseAction(DLNAServiceInfo& svc,

341 act.addArgument("StartingIndex", buf);
342 snprintf(buf, sizeof(buf), "%d", requestedCount);
343 act.addArgument("RequestedCount", buf);
344 act.addArgument("SortCriteria", "");
