133 if (available < 0) available = 0;

137 } else {

183 bool getData(int index, MediaItem& item, void* reference = nullptr) {
184 // Calculate the actual index into resultNodes_
185 int adjustedIndex = startingIndex_ + index;
186
187 // Validate index
188 if (adjustedIndex < 0 ||
189 adjustedIndex >= static_cast<int>(resultNodes_.size())) {
190 return false;
191 }
192
193 TreeNode* node = resultNodes_[adjustedIndex];
194 if (!node) return false;
195
196 // Populate MediaItem from TreeNode
197 // Use member strings for pointer lifetime
198 currentId_ = std::to_string(node->id);
199 item.id = currentId_.c_str();
200
201 // Parent ID
202 if (node->parent) {
203 currentParentId_ = std::to_string(node->parent->id);
204 item.parentID = currentParentId_.c_str();
205 } else {
206 item.parentID = "0";
207 }
208
209 item.title = node->file_name.c_str();
210 item.restricted = true;
211
212 if (node->is_dir) {
213 item.itemClass = MediaItemClass::Folder;
214 item.resourceURI = nullptr;
215 item.mimeType = nullptr;
216 } else {
217 // File - determine class from MIME type
218 currentMime_ = node->getMime();
219 if (currentMime_.empty()) currentMime_ = "application/octet-stream";
220 item.mimeType = currentMime_.c_str();
221
222 if (currentMime_.find("audio") != std::string::npos) {
223 item.itemClass = MediaItemClass::Music;
224 } else if (currentMime_.find("video") != std::string::npos) {
225 item.itemClass = MediaItemClass::Video;
226 } else if (currentMime_.find("image") != std::string::npos) {
227 item.itemClass = MediaItemClass::Photo;
228 } else {
229 item.itemClass = MediaItemClass::Unknown;
230 }
231
232 // Resource URI is the file path
233 currentPath_ = directoryTree.path(node->id);
234 item.resourceURI = currentPath_.c_str();
235 }
236
237 item.albumArtURI = nullptr;
238
239 return true;
240 }
