125 bool browse(int startingIndex, int requestedCount, XMLCallback XMLCallback,
126 int& numberReturned, int& totalMatches, int& updateID,
127 const char* browseFlag = nullptr) {
128
129 // Register item callback
130 this->mgr.onResultNode(XMLCallback);
131 // Build and post browse action
132 DLNAServiceInfo& svc =
133 selectService("urn:upnp-org:serviceId:ContentDirectory");
134 if (!svc) return false;
135 ActionRequest& act =
136 createBrowseAction(svc, browseFlag, startingIndex, requestedCount);
137 mgr.addAction(act);
138 last_reply = mgr.executeActions();
139 if (!last_reply) return false;
140
141 // parse numeric result fields
142 parseNumericFields(last_reply, numberReturned, totalMatches, updateID);
143
144 return true;
145 }

305 ActionRequest &createBrowseAction(DLNAServiceInfo& svc, const char* browseFlag,
306 int startingIndex, int requestedCount) {
307 static ActionRequest act(svc, "Browse");
308 // Use the canonical argument name expected by ContentDirectory: "ObjectID"
309 act.addArgument("ObjectID", object_id);
310 act.addArgument("BrowseFlag",
311 browseFlag ? browseFlag : "BrowseDirectChildren");
312 act.addArgument("Filter", "");
313 char buf[32];
314 snprintf(buf, sizeof(buf), "%d", startingIndex);
315 act.addArgument("StartingIndex", buf);
316 snprintf(buf, sizeof(buf), "%d", requestedCount);
317 act.addArgument("RequestedCount", buf);
318 act.addArgument("SortCriteria", "");
319 return act;
320 }
