264 // Transport control handler
265 static void transportControlCB(HttpServer* server, const char* requestPath,
266 HttpRequestHandlerLine* hl) {
267 // Parse SOAP request and extract action
268 Str soap = server->contentStr();
269 DlnaLogger.log(DlnaLogLevel::Info, "Transport Control: %s", soap.c_str());
270 MediaRenderer& media_renderer = *((MediaRenderer*)hl->context[0]);
271
272 Str reply_str{reply()};
273 reply_str.replaceAll("%2", "AVTransport");
274 if (soap.indexOf("Play") >= 0) {
275 media_renderer.setActive(true);
276 reply_str.replaceAll("%1", "PlayResponse");
277 server->reply("text/xml", reply_str.c_str());
278 } else if (soap.indexOf("Pause") >= 0) {
279 media_renderer.setActive(false);
280 reply_str.replaceAll("%1", "PauseResponse");
281 server->reply("text/xml", reply_str.c_str());
282 } else if (soap.indexOf("Stop") >= 0) {
283 media_renderer.setActive(false);
284 reply_str.replaceAll("%1", "StopResponse");
285 server->reply("text/xml", reply_str.c_str());
286 } else if (soap.indexOf("SetAVTransportURI") >= 0) {
287 // Extract URL from SOAP request safely
288 int urlTagStart = soap.indexOf("<CurrentURI>");
289 int urlTagEnd = soap.indexOf("</CurrentURI>");
290 if (urlTagStart >= 0 && urlTagEnd > urlTagStart) {
291 int urlStart = urlTagStart + (int)strlen("<CurrentURI>");
292 Str url = soap.substring(urlStart, urlTagEnd);
293 media_renderer.play(url.c_str());
294 } else {
295 DlnaLogger.log(DlnaLogLevel::Warning,
296 "SetAVTransportURI called with invalid SOAP payload");
297 }
298 reply_str.replaceAll("%1", "SetAVTransportURIResponse");
299 server->reply("text/xml", reply_str.c_str());
300 } else {
301 // Handle other transport controls
302 reply_str.replaceAll("%1", "ActionResponse");
303 server->reply("text/xml", reply_str.c_str());
304 }
305 }

308 static void renderingControlCB(HttpServer* server, const char* requestPath,
309 HttpRequestHandlerLine* hl) {
310 // Parse SOAP request and extract action
311 MediaRenderer& media_renderer = *((MediaRenderer*)hl->context[0]);
312 Str soap = server->contentStr();
313 DlnaLogger.log(DlnaLogLevel::Info, "Rendering Control: %s", soap.c_str());
314 Str reply_str{reply()};
315 reply_str.replaceAll("%2", "RenderingControl");
316
317 if (soap.indexOf("SetVolume") >= 0) {
318 // Extract volume from SOAP request safely
319 int volTagStart = soap.indexOf("<DesiredVolume>");
320 int volTagEnd = soap.indexOf("</DesiredVolume>");
321 if (volTagStart >= 0 && volTagEnd > volTagStart) {
322 int volStart = volTagStart + (int)strlen("<DesiredVolume>");
323 int volume = soap.substring(volStart, volTagEnd).toInt();
324 media_renderer.setVolume(volume);
325 } else {
326 DlnaLogger.log(DlnaLogLevel::Warning,
327 "SetVolume called with invalid SOAP payload");
328 }
329 reply_str.replaceAll("%1", "SetVolumeResponse");
330 server->reply("text/xml", reply_str.c_str());
331 } else if (soap.indexOf("SetMute") >= 0) {
332 // Extract mute state from SOAP request safely
333 int muteTagStart = soap.indexOf("<DesiredMute>");
334 int muteTagEnd = soap.indexOf("</DesiredMute>");
335 if (muteTagStart >= 0 && muteTagEnd > muteTagStart) {
336 int muteStart = muteTagStart + (int)strlen("<DesiredMute>");
337 bool mute = (soap.substring(muteStart, muteTagEnd) == "1");
338 media_renderer.setMute(mute);
339 } else {
340 DlnaLogger.log(DlnaLogLevel::Warning,
341 "SetMute called with invalid SOAP payload");
342 }
343 reply_str.replaceAll("%1", "SetMuteResponse");
344 server->reply("text/xml", reply_str.c_str());
345 } else {
346 // Handle other rendering controls
347 reply_str.replaceAll("%1", "RenderingControl");
348 server->reply("text/xml", reply_str.c_str());
