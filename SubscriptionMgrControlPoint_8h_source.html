421 bool subscribeToService(DLNAServiceInfo& service) {
422 if (StrView(service.event_sub_url).isEmpty()) {
423 return false;
424 }
425 Url url(service.event_sub_url);
426
427 // If already subscribed and not expired, nothing to do
428 if (service.subscription_state == SubscriptionState::Subscribed &&
429 millis() < service.time_subscription_expires) {
430 return true;
431 }
432
433 char seconds_txt[80] = {0};
434 snprintf(seconds_txt, sizeof(seconds_txt), "Second-%d",
435 event_subscription_duration_sec);
436 p_http->request().put("NT", "upnp:event");
437 p_http->request().put("TIMEOUT", seconds_txt);
438 p_http->request().put("CALLBACK", p_local_url ? p_local_url->url() : "");
439 // For re-subscribe, include existing SID header if present
440 if (!service.subscription_id.isEmpty()) {
441 p_http->request().put("SID", service.subscription_id.c_str());
442 }
443
444 int rc = p_http->subscribe(url);
445 if (rc == 200) {
446 const char* sid = p_http->reply().get("SID");
447 if (sid != nullptr) {
448 service.subscription_id = sid;
449 service.subscription_state = SubscriptionState::Subscribed;
450 service.time_subscription_started = millis();
451 service.time_subscription_confirmed = millis();
452 service.time_subscription_expires =
453 millis() + (event_subscription_duration_sec * 1000);
454 subscribe_timeout = service.time_subscription_expires;
455 DlnaLogger.log(DlnaLogLevel::Info, "Subscribe %s -> rc=%d", url.url(),
456 rc);
457 } else {
458 DlnaLogger.log(DlnaLogLevel::Warning,
459 "Subscribe %s succeeded but no SID returned", url.url());
460 return false;
461 }
462 } else {
463 // failed to subscribe
464 DlnaLogger.log(DlnaLogLevel::Error,
465 "Failed to subscribe to service %s, rc=%d",
466 service.service_id, rc);
467 return false;
468 }
469 return true;
470 }

483 bool unsubscribeFromService(DLNAServiceInfo& service) {
484 if (StrView(service.event_sub_url).isEmpty()) {
485 return false;
486 }
487 // If already subscribed and not expired, nothing to do
488 if (service.subscription_state == SubscriptionState::Unsubscribed) {
489 return true;
490 }
491
492 Url url(service.event_sub_url);
493 int rc = p_http->unsubscribe(url, service.event_sub_sid);
494 if (rc == 200) {
495 DlnaLogger.log(DlnaLogLevel::Info, "Unsubscribe %s -> rc=%d", url.url(),
496 rc);
497 // clear SID and metadata regardless of remote rc
498 service.event_sub_sid = nullptr;
499 service.subscription_id = "";
500 service.subscription_state = SubscriptionState::Unsubscribed;
501 service.time_subscription_confirmed = 0;
502 service.time_subscription_expires = 0;
503 subscribe_timeout = 0;
504 } else {
505 DlnaLogger.log(DlnaLogLevel::Error,
506 "Failed to unsubscribe from service %s, rc=%d",
507 service.service_id, rc);
508 return false;
509 }
510 return true;

533 static void notifyHandler(HttpServer* server, const char* requestPath,
534 HttpRequestHandlerLine* handlerLine) {
535 if (handlerLine == nullptr || handlerLine->contextCount < 1) return;
536 void* ctx0 = handlerLine->context[0];
537 SubscriptionMgrControlPoint* mgr =
538 static_cast<SubscriptionMgrControlPoint*>(ctx0);
539 if (!mgr) return;
540 void* reference = mgr->event_callback_ref;
541
542 // read SID header and body
543 const char* sid = server->requestHeader().get("SID");
544
545 // Use XMLParserPrint to incrementally parse the NOTIFY body and extract
546 // property child elements. XMLParserPrint accumulates the buffer and
547 // exposes a parse() method that returns the next fragment.
548
549 // mark notify received early so subscription metadata gets updated
550 mgr->updateReceived(sid);
551
552 XMLParserPrint xml_parser;
553 uint8_t buffer[XML_PARSER_BUFFER_SIZE];
554 Str nodeName;
555 Vector<Str> path;
556 Str text;
557 Str attrs;
558 xml_parser.setExpandEncoded(true);
559 Client& client = server->client();
560
561 while (client.available()) {
562 int read = client.read(buffer, XML_PARSER_BUFFER_SIZE);
563 xml_parser.write(buffer, read);
564
565 while (xml_parser.parse(nodeName, path, text, attrs)) {
566 // We care about text nodes that are children of a <property> element
567 if (text.isEmpty()) continue;
568
569 // dispatch
570 DLNAServiceInfo& s = mgr->getServiceInfoBySID(sid);
571 if (mgr->event_callback) {
572 const char* sid_c = sid ? sid : "";
573 mgr->event_callback(sid_c, nodeName.c_str(), text.c_str(), reference);
574 }
575 }
576 }
577
578 // reply OK to the NOTIFY
579 server->replyOK();
580 }
