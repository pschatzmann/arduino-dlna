356 bool processAction(ActionRequest& action, HttpServer& server) {
357 DlnaLogger.log(DlnaLogLevel::Info, "DLNAMediaServer::processAction: %s",
358 action.action);
359 StrView action_str(action.action);
360 if (action_str.isEmpty()) {
361 DlnaLogger.log(DlnaLogLevel::Error, "Empty action received");
362 server.replyNotFound();
363 return false;
364 }
365 for (const auto& rule : rules) {
366 if (action_str.endsWith(rule.suffix)) {
367 return rule.handler(this, action, server);
368 }
369 }
370 DlnaLogger.log(DlnaLogLevel::Error, "Unsupported action: %s",
371 action.action);
372 server.replyNotFound();
373 return false;
374 }

623 bool streamActionItems(const char* responseName, const char* objectID,
624 ContentQueryType queryType, const char* filter,
625 int startingIndex, int requestedCount,
626 HttpServer& server) {
627 int numberReturned = 0;
628 int totalMatches = 0;
629 int updateID = g_stream_updateID;
630
631 // allow caller to prepare counts/updateID
632 if (prepare_data_cb) {
633 prepare_data_cb(objectID, queryType, filter, startingIndex,
634 requestedCount, nullptr, numberReturned, totalMatches,
635 updateID, reference_);
636 }
637
638 ChunkPrint chunk_writer{server.client()};
639 server.replyChunked("text/xml");
640
641 // SOAP envelope start and response wrapper
642 soapEnvelopeStart(chunk_writer);
643 // responseName is assumed to be e.g. "BrowseResponse" or "SearchResponse"
644 actionResponseStart(chunk_writer, responseName,
645 "urn:schemas-upnp-org:service:ContentDirectory:1");
646
647 // Result -> DIDL-Lite
648 chunk_writer.println("<Result>");
649 // DIDL root with namespaces
650 streamDIDL(chunk_writer, numberReturned, startingIndex);
651 chunk_writer.print("</Result>\r\n");
652
653 // numeric fields
654 chunk_writer.printf("<NumberReturned>%d</NumberReturned>\r\n",
655 numberReturned);
656 chunk_writer.printf("<TotalMatches>%d</TotalMatches>\r\n", totalMatches);
657 chunk_writer.printf("<UpdateID>%d</UpdateID>\r\n", updateID);
658
659 actionResponseEnd(chunk_writer, responseName);
660 soapEnvelopeEnd(chunk_writer);
661 chunk_writer.end();
662 return true;
663 }
664
666 void streamDIDL(ChunkPrint& chunk_writer, int numberReturned,
667 int startingIndex) {
668 // Stream DIDL-Lite Result payload using helper
669 chunk_writer.print(
670 "&lt;DIDL-Lite xmlns:dc=\"http://purl.org/dc/elements/1.1/\" "
671 "xmlns:upnp=\"urn:schemas-upnp-org:metadata-1-0/upnp/\" "
672 "xmlns=\"urn:schemas-upnp-org:metadata-1-0/DIDL-Lite/\"&gt;\r\n");
673
674 if (get_data_cb) {
675 Str url{160};
676 for (int i = 0; i < numberReturned; ++i) {
677 MediaItem item;
678 int idx = startingIndex + i;
679 if (!get_data_cb(idx, item, reference_)) break;
680
681 const char* mediaItemClassStr = toStr(item.itemClass);
682 const char* nodeName =
683 (item.itemClass == MediaItemClass::Folder) ? "container" : "item";
684
685 chunk_writer.printf(
686 "&lt;%s id=\"%s\" parentID=\"%s\" restricted=\"%d\"&gt;", nodeName,
687 item.id ? item.id : "", item.parentID ? item.parentID : "0",
688 item.restricted ? 1 : 0);
689
690 // title
691 chunk_writer.print("&lt;dc:title&gt;");
692 chunk_writer.print(StringRegistry::nullStr(item.title));
693 chunk_writer.print("&lt;/dc:title&gt;\r\n");
694 if (mediaItemClassStr != nullptr) {
695 chunk_writer.printf("&lt;upnp:class&gt;%s&lt;/upnp:class&gt;\r\n",
696 mediaItemClassStr);
697 }
698
699 // res with optional protocolInfo attribute
700 url.set(item.resourceURL);
701 if (!url.isEmpty()) {
702 url.replaceAll("&", "&amp;");
703 if (!StrView(item.mimeType).isEmpty()) {
704 chunk_writer.printf("&lt;res protocolInfo=\"http-get:*:%s:*\"&gt;",
705 item.mimeType);
706 chunk_writer.print(StringRegistry::nullStr(url.c_str()));
707 chunk_writer.print("&lt;/res&gt;\r\n");
708 } else {
709 chunk_writer.print("&lt;res&gt;");
710 chunk_writer.print(StringRegistry::nullStr(url.c_str()));
711 chunk_writer.print("&lt;/res&gt;\r\n");
712 }
713 }
714
715 chunk_writer.printf("&lt;/%s&gt;\r\n", nodeName);
716 }
717 }
718
719 chunk_writer.print("&lt;/DIDL-Lite&gt;\r\n");
720 }

739 // Control handler for ContentDirectory service
740 static void contentDirectoryControlCB(HttpServer* server,
741 const char* requestPath,
742 HttpRequestHandlerLine* hl) {
743 ActionRequest action;
744 DLNADevice::parseActionRequest(server, requestPath, hl, action);
745
746 // If a user-provided callback is registered, hand over the parsed
747 // ActionRequest for custom handling. The callback is responsible for
748 // writing the HTTP reply if it handles the action.
749 DLNAMediaServer* ms = nullptr;
750 if (hl && hl->context[0]) ms = (DLNAMediaServer*)hl->context[0];
751
752 // process the requested action using instance method if available
753 if (ms) {
754 if (ms->processAction(action, *server)) return;
755 }
756
757 server->replyNotFound();
758 }

760 // simple connection manager control that replies OK
761 static void connmgrControlCB(HttpServer* server, const char* requestPath,
762 HttpRequestHandlerLine* hl) {
763 ActionRequest action;
764 DLNADevice::parseActionRequest(server, requestPath, hl, action);
765
766 DLNAMediaServer* ms = nullptr;
767 if (hl && hl->context[0]) ms = (DLNAMediaServer*)hl->context[0];
768 if (!ms) {
769 server->replyNotFound();
770 return;
771 }
772
773 // Use rules-based dispatch for ConnectionManager actions
774 if (ms->processAction(action, *server)) return;
775 server->replyNotFound();
776 }
