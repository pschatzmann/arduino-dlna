342 static void eventSubscriptionHandler(HttpServer* server,

345 DLNADevice::eventSubscriptionHandler(server, requestPath, hl);

363 DLNAServiceInfo cd;
364 cd.setup("urn:schemas-upnp-org:service:ContentDirectory:1",
365 "urn:upnp-org:serviceId:ContentDirectory", "/CD/service.xml",
366 &DLNAMediaServer::contentDescCB, "/CD/control",

370
371 // subscription namespace abbreviation used for event publishing

374 // register URLs on the provided server so SCPD, control and event

379 server->on(cd.control_url, T_POST, cd.control_cb);

381 server->on(cd.event_sub_url, T_UNSUBSCRIBE,
382 tiny_dlna::DLNADevice::eventSubscriptionHandler);

411 bool processAction(ActionRequest& action, HttpServer& server) {
412 DlnaLogger.log(DlnaLogLevel::Info, "DLNAMediaServer::processAction: %s",
413 StringRegistry::nullStr(action.action));
414 StrView action_str(action.action);
415 if (action_str.isEmpty()) {
416 DlnaLogger.log(DlnaLogLevel::Error, "Empty action received");
417 server.replyNotFound();
418 return false;
419 }
420 for (const auto& rule : rules) {
421 if (action_str.endsWith(rule.suffix)) {
422 return rule.handler(this, action, server);
423 }
424 }
425 DlnaLogger.log(DlnaLogLevel::Error, "Unsupported action: %s",
426 action.action);
427 server.replyNotFound();
428 return false;
429 }

432 bool processActionBrowse(ActionRequest& action, HttpServer& server) {
433 DlnaLogger.log(DlnaLogLevel::Info, "processActionBrowse");
434 int numberReturned = 0;
435 int totalMatches = 0;
436 int updateID = g_stream_updateID;
437
438 // extract paging args
439 int startingIndex = action.getArgumentIntValue("StartingIndex");
440 int requestedCount = action.getArgumentIntValue("RequestedCount");
441
442 // determine query type (BrowseDirectChildren / BrowseMetadata / Search)
443 ContentQueryType qtype =
444 parseContentQueryType(action.getArgumentValue("BrowseFlag"));
445 // query for data
446 if (prepare_data_cb) {
447 prepare_data_cb(action.getArgumentValue("ObjectID"), qtype,
448 action.getArgumentValue("Filter"), startingIndex,
449 requestedCount, action.getArgumentValue("SortCriteria"),
450 numberReturned, totalMatches, updateID, reference_);
451 }
452
453 // provide result
454 return streamActionItems("BrowseResponse",
455 action.getArgumentValue("ObjectID"), qtype,
456 action.getArgumentValue("Filter"), startingIndex,
457 requestedCount, server);
458 }

624 bool streamActionItems(const char* responseName, const char* objectID,
625 ContentQueryType queryType, const char* filter,
626 int startingIndex, int requestedCount,
627 HttpServer& server) {
628 int numberReturned = 0;
629 int totalMatches = 0;
630 int updateID = g_stream_updateID;
631
632 // allow caller to prepare counts/updateID
633 if (prepare_data_cb) {
634 prepare_data_cb(objectID, queryType, filter, startingIndex,
635 requestedCount, nullptr, numberReturned, totalMatches,
636 updateID, reference_);
637 }
638
639 ChunkPrint chunk_writer{server.client()};
640 server.replyChunked("text/xml");
641
642 // SOAP envelope start and response wrapper
643 soapEnvelopeStart(chunk_writer);
644 // responseName is assumed to be e.g. "BrowseResponse" or "SearchResponse"
645 actionResponseStart(chunk_writer, responseName,
646 "urn:schemas-upnp-org:service:ContentDirectory:1");
647
648 // Result -> DIDL-Lite
649 chunk_writer.println("<Result>");
650 // DIDL root with namespaces
651 streamDIDL(chunk_writer, numberReturned, startingIndex);
652 chunk_writer.print("</Result>\r\n");
653
654 // numeric fields
655 chunk_writer.printf("<NumberReturned>%d</NumberReturned>\r\n",
656 numberReturned);
657 chunk_writer.printf("<TotalMatches>%d</TotalMatches>\r\n", totalMatches);
658 chunk_writer.printf("<UpdateID>%d</UpdateID>\r\n", updateID);
659
660 actionResponseEnd(chunk_writer, responseName);
661 soapEnvelopeEnd(chunk_writer);
662 chunk_writer.end();
663 return true;
664 }

667 void streamDIDL(ChunkPrint& chunk_writer, int numberReturned,
668 int startingIndex) {
669 // Stream DIDL-Lite Result payload using helper
670 chunk_writer.print(
671 "&lt;DIDL-Lite xmlns:dc=\"http://purl.org/dc/elements/1.1/\" "
672 "xmlns:upnp=\"urn:schemas-upnp-org:metadata-1-0/upnp/\" "
673 "xmlns=\"urn:schemas-upnp-org:metadata-1-0/DIDL-Lite/\"&gt;\r\n");
674
675 if (get_data_cb) {
676 Str url{160};
677 for (int i = 0; i < numberReturned; ++i) {
678 MediaItem item;
679 int idx = startingIndex + i;
680 if (!get_data_cb(idx, item, reference_)) break;
681
682 const char* mediaItemClassStr = toStr(item.itemClass);
683 const char* nodeName =
684 (item.itemClass == MediaItemClass::Folder) ? "container" : "item";
685
686 chunk_writer.printf(
687 "&lt;%s id=\"%s\" parentID=\"%s\" restricted=\"%d\"&gt;", nodeName,
688 item.id ? item.id : "", item.parentID ? item.parentID : "0",
689 item.restricted ? 1 : 0);
690
691 // title
692 chunk_writer.print("&lt;dc:title&gt;");
693 chunk_writer.print(StringRegistry::nullStr(item.title));
694 chunk_writer.print("&lt;/dc:title&gt;\r\n");
695 if (mediaItemClassStr != nullptr) {
696 chunk_writer.printf("&lt;upnp:class&gt;%s&lt;/upnp:class&gt;\r\n",
697 mediaItemClassStr);
698 }
699
700 // res with optional protocolInfo attribute
701 url.set(item.resourceURL);
702 if (!url.isEmpty()) {
703 url.replaceAll("&", "&amp;");
704 if (!StrView(item.mimeType).isEmpty()) {
705 chunk_writer.printf("&lt;res protocolInfo=\"http-get:*:%s:*\"&gt;",
706 item.mimeType);
707 chunk_writer.print(StringRegistry::nullStr(url.c_str()));
708 chunk_writer.print("&lt;/res&gt;\r\n");
709 } else {
710 chunk_writer.print("&lt;res&gt;");
711 chunk_writer.print(StringRegistry::nullStr(url.c_str()));
712 chunk_writer.print("&lt;/res&gt;\r\n");
713 }
714 }
715
716 chunk_writer.printf("&lt;/%s&gt;\r\n", nodeName);
717 }
718 }
719
720 chunk_writer.print("&lt;/DIDL-Lite&gt;\r\n");
721 }
