403 bool processAction(ActionRequest& action, HttpServer& server) {
404 DlnaLogger.log(DlnaLogLevel::Info, "DLNAMediaServer::processAction: %s",
405 StringRegistry::nullStr(action.action));
406 StrView action_str(action.action);
407 if (action_str.isEmpty()) {
408 DlnaLogger.log(DlnaLogLevel::Error, "Empty action received");
409 server.replyNotFound();
410 return false;
411 }
412 for (const auto& rule : rules) {
413 if (action_str.endsWith(rule.suffix)) {
414 return rule.handler(this, action, server);
415 }
416 }
417 DlnaLogger.log(DlnaLogLevel::Error, "Unsupported action: %s",
418 action.action);
419 server.replyNotFound();
420 return false;
421 }

615 bool streamActionItems(const char* responseName, const char* objectID,
616 ContentQueryType queryType, const char* filter,
617 int startingIndex, int requestedCount,
618 HttpServer& server) {
619 int numberReturned = 0;
620 int totalMatches = 0;
621 int updateID = g_stream_updateID;
622
623 // allow caller to prepare counts/updateID
624 if (prepare_data_cb) {
625 prepare_data_cb(objectID, queryType, filter, startingIndex,
626 requestedCount, nullptr, numberReturned, totalMatches,
627 updateID, reference_);
628 }
629
630 ChunkPrint chunk_writer{server.client()};
631 server.replyChunked("text/xml");
632
633 // SOAP envelope start and response wrapper
634 soapEnvelopeStart(chunk_writer);
635 // responseName is assumed to be e.g. "BrowseResponse" or "SearchResponse"
636 actionResponseStart(chunk_writer, responseName,
637 "urn:schemas-upnp-org:service:ContentDirectory:1");
638
639 // Result -> DIDL-Lite
640 chunk_writer.println("<Result>");
641 // DIDL root with namespaces
642 streamDIDL(chunk_writer, numberReturned, startingIndex);
643 chunk_writer.print("</Result>\r\n");
644
645 // numeric fields
646 chunk_writer.printf("<NumberReturned>%d</NumberReturned>\r\n",
647 numberReturned);
648 chunk_writer.printf("<TotalMatches>%d</TotalMatches>\r\n", totalMatches);
649 chunk_writer.printf("<UpdateID>%d</UpdateID>\r\n", updateID);
650
651 actionResponseEnd(chunk_writer, responseName);
652 soapEnvelopeEnd(chunk_writer);
653 chunk_writer.end();
654 return true;
655 }

658 void streamDIDL(ChunkPrint& chunk_writer, int numberReturned,
659 int startingIndex) {
660 // Stream DIDL-Lite Result payload using helper
661 chunk_writer.print(
662 "&lt;DIDL-Lite xmlns:dc=\"http://purl.org/dc/elements/1.1/\" "
663 "xmlns:upnp=\"urn:schemas-upnp-org:metadata-1-0/upnp/\" "
664 "xmlns=\"urn:schemas-upnp-org:metadata-1-0/DIDL-Lite/\"&gt;\r\n");
665
666 if (get_data_cb) {
667 Str url{160};
668 for (int i = 0; i < numberReturned; ++i) {
669 MediaItem item;
670 int idx = startingIndex + i;
671 if (!get_data_cb(idx, item, reference_)) break;
672
673 const char* mediaItemClassStr = toStr(item.itemClass);
674 const char* nodeName =
675 (item.itemClass == MediaItemClass::Folder) ? "container" : "item";
676
677 chunk_writer.printf(
678 "&lt;%s id=\"%s\" parentID=\"%s\" restricted=\"%d\"&gt;", nodeName,
679 item.id ? item.id : "", item.parentID ? item.parentID : "0",
680 item.restricted ? 1 : 0);
681
682 // title
683 chunk_writer.print("&lt;dc:title&gt;");
684 chunk_writer.print(StringRegistry::nullStr(item.title));
685 chunk_writer.print("&lt;/dc:title&gt;\r\n");
686 if (mediaItemClassStr != nullptr) {
687 chunk_writer.printf("&lt;upnp:class&gt;%s&lt;/upnp:class&gt;\r\n",
688 mediaItemClassStr);
689 }
690
691 // res with optional protocolInfo attribute
692 url.set(item.resourceURL);
693 if (!url.isEmpty()) {
694 url.replaceAll("&", "&amp;");
695 if (!StrView(item.mimeType).isEmpty()) {
696 chunk_writer.printf("&lt;res protocolInfo=\"http-get:*:%s:*\"&gt;",
697 item.mimeType);
698 chunk_writer.print(StringRegistry::nullStr(url.c_str()));
699 chunk_writer.print("&lt;/res&gt;\r\n");
700 } else {
701 chunk_writer.print("&lt;res&gt;");
702 chunk_writer.print(StringRegistry::nullStr(url.c_str()));
703 chunk_writer.print("&lt;/res&gt;\r\n");
704 }
705 }
706
707 chunk_writer.printf("&lt;/%s&gt;\r\n", nodeName);
708 }
709 }
710
711 chunk_writer.print("&lt;/DIDL-Lite&gt;\r\n");
712 }
