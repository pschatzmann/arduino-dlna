413 bool processAction(ActionRequest& action, IHttpServer& server) {
414 DlnaLogger.log(DlnaLogLevel::Info, "DLNAMediaServer::processAction: %s",
415 action.getAction());
416 auto& action_str = action.getActionStr();
417 if (action_str.isEmpty()) {
418 DlnaLogger.log(DlnaLogLevel::Error, "Empty action received");
419 server.replyNotFound();
420 return false;
421 }
422 for (const auto& rule : rules) {
423 if (action_str.endsWith(rule.suffix)) {
424 return rule.handler(this, action, server);
425 }
426 }
427 DlnaLogger.log(DlnaLogLevel::Error, "Unsupported action: %s",
428 action.getAction());
429 server.replyNotFound();
430 return false;
431 }

435 DlnaLogger.log(DlnaLogLevel::Info, "processActionBrowse");
436 int numberReturned = 0;
437 int totalMatches = 0;
438 int updateID = g_stream_updateID;
439
440 // extract paging args
441 int startingIndex = action.getArgumentIntValue("StartingIndex");
442 int requestedCount = action.getArgumentIntValue("RequestedCount");
443
444 // determine query type (BrowseDirectChildren / BrowseMetadata / Search)
445 ContentQueryType qtype =
446 parseContentQueryType(action.getArgumentValue("BrowseFlag"));
447 // query for data
448 if (prepare_data_cb) {
449 prepare_data_cb(action.getArgumentValue("ObjectID"), qtype,
450 action.getArgumentValue("Filter"), startingIndex,

453 }

464 auto self = (DLNAMediaServer*)ref;

466 self->g_stream_startingIndex);
467 },
468 200, nullptr, this);

679 size_t streamActionItems(Print& out, const char* responseName,
680 int startingIndex) {
681 // SOAP envelope start and response wrapper
682 size_t written = 0;
683 written += soapEnvelopeStart(out);
684 written += actionResponseStart(
685 out, responseName, "urn:schemas-upnp-org:service:ContentDirectory:1");
686
687 // Result -> DIDL-Lite
688 written += out.println("<Result>");
689 // DIDL root with namespaces
690 written += streamDIDL(out, g_stream_numberReturned, startingIndex);
691 written += out.print("</Result>\r\n");

693 // numeric fields
694 Printf pr{out};
695 written += static_cast<size_t>(pr.printf(

697 written += static_cast<size_t>(pr.printf(
698 "<TotalMatches>%d</TotalMatches>\r\n", g_stream_totalMatches));
699 written += static_cast<size_t>(
700 pr.printf("<UpdateID>%d</UpdateID>\r\n", g_stream_updateID));
701
702 written += actionResponseEnd(out, responseName);
703 written += soapEnvelopeEnd(out);
704 return written;

708 size_t streamDIDL(Print& out, int numberReturned, int startingIndex) {
709 size_t written = 0;
710 // Stream DIDL-Lite Result payload using helper
711 written += out.print(
712 "&lt;DIDL-Lite xmlns:dc=\"http://purl.org/dc/elements/1.1/\" "
713 "xmlns:upnp=\"urn:schemas-upnp-org:metadata-1-0/upnp/\" "
714 "xmlns=\"urn:schemas-upnp-org:metadata-1-0/DIDL-Lite/\"&gt;\r\n");

716 written += streamDIDLItems(out, numberReturned, startingIndex);
717
718 written += out.print("&lt;/DIDL-Lite&gt;\r\n");

722 // Streams each MediaItem entry (container/item with title,
723 // class and optional resource). Returns bytes written.
724 // Single-pass loop that prefers get_data_print_cb and falls back
725 // to get_data_cb for any item where print produced 0 bytes.
726 size_t streamDIDLItems(Print& out, int numberReturned, int startingIndex) {
727 if (get_data_print_cb == nullptr && get_data_cb == nullptr) {
728 // no data callback defined
729 return 0;
730 }
731 EscapingPrint esc_out(out);
732 size_t total = 0;
733 for (int i = 0; i < numberReturned; ++i) {
734 int idx = startingIndex + i;
735
736 // 1) Try print callback if present
737 if (get_data_print_cb) {
738 size_t w = get_data_print_cb(idx, esc_out, reference_);
739 if (w > 0) {
740 total += w;
741 continue; // done with this item
742 }
743 }
744
745 // 2) Structured callback path
746 if (get_data_cb) {
747 MediaItem item;
748 if (!get_data_cb(idx, item, reference_)) break; // end-of-list
749 total += streamDIDLItem(esc_out, item);
750 continue;
751 }
752 }
753 return total;
754 }

756 // Print a single MediaItem as escaped DIDL (returns bytes written)
757 size_t streamDIDLItem(Print& out, const MediaItem& item) {
758 size_t written = 0;
759 const char* mediaItemClassStr = toStr(item.itemClass);
760 const char* nodeName =
761 (item.itemClass == MediaItemClass::Folder) ? "container" : "item";
762 Printf pr{out};
763 written += static_cast<size_t>(
764 pr.printf("<%s id=\"%s\" parentID=\"%s\" restricted=\"%d\">", nodeName,
765 item.id ? item.id : "", item.parentID ? item.parentID : "0",
766 item.restricted ? 1 : 0));
767
768 written += out.print("<dc:title>");
769 written += out.print(StrView(item.title).c_str());
770 written += out.print("</dc:title>\r\n");
771 if (mediaItemClassStr != nullptr) {
772 written += static_cast<size_t>(
773 pr.printf("<upnp:class>%s</upnp:class>\r\n", mediaItemClassStr));
774 }
775
776 // Optional album art URI
777 if (!StrView(item.albumArtURI).isEmpty()) {
778 Str url_str = getUri(item.albumArtURI);
779 written += out.print("<upnp:albumArtURI>");
780 written += out.print(url_str.c_str());
781 written += out.print("</upnp:albumArtURI>\r\n");
782 }
783
784 // res with optional protocolInfo attribute
785 if (!StrView(item.resourceURI).isEmpty()) {
786 if (!StrView(item.mimeType).isEmpty()) {
787 written += static_cast<size_t>(
788 pr.printf("<res protocolInfo=\"http-get:*:%s:*\">", item.mimeType));
789 } else {
790 written += out.print("<res>");
791 }
792 Str url_str = getUri(item.resourceURI);
793 written += out.print(url_str.c_str());
794 written += out.print("</res>\r\n");
795 }
796
797 written += static_cast<size_t>(pr.printf("</%s>\r\n", nodeName));
798 return written;
799 }

802 Str getUri(const char* path) {
803 Str url{256};
804 // We got a valid url
805 url = path;
806 if (url.startsWith("http://")) {
807 return url;
808 }
809 // We need to add the path to the base url
810 url = getBaseURL();
811 if (!StrView(path).startsWith("/")) {
812 url += "/";
813 }
814 url += path;
815 return url;
816 }
