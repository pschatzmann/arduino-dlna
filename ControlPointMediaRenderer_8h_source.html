458 // Helper: parse DLNA time string (HH:MM:SS, MM:SS) to milliseconds
459 static unsigned long parseTimeToMs(const char* t) {
460 if (t == nullptr) return 0;
461 int h = 0, m = 0, s = 0;
462 int parts = 0;
463 // Count colons
464 const char* p = t;
465 int colonCount = 0;
466 while (*p) {
467 if (*p == ':') colonCount++;
468 p++;
469 }
470 if (colonCount == 2) {
471 parts = sscanf(t, "%d:%d:%d", &h, &m, &s);
472 } else if (colonCount == 1) {
473 parts = sscanf(t, "%d:%d", &m, &s);
474 h = 0;
475 } else {
476 // maybe it's seconds only
477 parts = sscanf(t, "%d", &s);
478 }
479 if (parts <= 0) return 0;
480 unsigned long total =
481 (unsigned long)h * 3600UL + (unsigned long)m * 60UL + (unsigned long)s;
482 return total * 1000UL;
483 }
